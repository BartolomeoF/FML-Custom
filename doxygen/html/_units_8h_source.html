103  void init(std::string type){
104 
105  if(type == "User units"){
106  description = "User units";
107  // The m, s, kg, K, Co constants
108  // have already been set so continue
109  } else if(type == "SI") {
110  description = "SI units";
111  m = 1.0;
112  s = 1.0;
113  kg = 1.0;
114  K = 1.0;
115  Co = 1.0;
116  } else if(type == "Planck") {
117  description = "Planck Units (c=hbar=G=kb=1)";
118  m = sqrt((c_SI * c_SI * c_SI) / (hbar_SI * G_SI));
119  s = c_SI * m;
120  kg = G_SI * (m*m*m) / (s*s);
121  K = k_b_SI * kg * (m*m) / (s*s);
122  Co = sqrt(k_e_SI * kg * (m*m*m) / (s*s));
123  } else if(type == "ParticlePhysics") {
124  description = "Particle Physics Units (c=hbar=kb=1) with eV = 1";
125  m = eV_SI / hbar_SI / c_SI;
126  s = c_SI * m;
127  kg = 1.0 / hbar_SI * s / (m*m);
128  K = k_b_SI * G_SI * (m*m*m) / (s*s) * (m*m) / (s*s);
129  Co = 1.0;
130  } else if(type == "Cosmology"){
131  description = "Cosmo Units (Mpc, Hubble time, Solar-masses, Kelvin)";
132  m = 1.0 / (1e6 * pc_SI); // Megaparsecs
133  s = 1e5 / (1e6 * pc_SI); // Hubble-time 1/(H0/h)
134  kg = 1.0 / Msun_SI; // Solar masses
135  K = 1.0; // Kelvin
136  Co = 1.0; // Coulomb
137  } else {
138  std::string errormessage = "[ConstantsAndUnits::init] Unknown units: "
139  + type + " Choose between: SI, Planck, ParticlePhysics, Cosmology\n";
140  throw_error(errormessage);
141  }

157  density = kg/(m*m*m); // Density

160  k_b = k_b_SI * J / K; // Bolzmanns constant
161  k_e = k_e_SI * N * m * m / (Co * Co); // Columbs constant
162  G = G_SI * N * m * m / (kg * kg); // Gravitational constant
163  hbar = hbar_SI * J * s; // Reduced Plancks constant
164  c = c_SI * m / s; // Speed of light
165  m_e = m_e_SI * kg; // Mass of electron
166  m_H = m_H_SI * kg; // Mass of hydrogen atom
