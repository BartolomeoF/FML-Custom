86  void ODESolver::solve(ODEFunctionPointer ode_equation,

88  DVector& xarr,
89  DVector& yinitial,
90  const gsl_odeiv2_step_type* stepper,
91  ODEFunctionPointerJacobian jacobian)
92  {
93  // Store the number of equations and size of x-array
94  nequations = yinitial.size();
95  num_x_points = xarr.size();
96  if (num_x_points < 2) {
97  std::string errormessage = "[ODESolver::solve] The xarray needs to have atleast size 2\n";
98  throw_error(errormessage);
99  }
100  if (nequations < 1) {
101  std::string errormessage = "[ODESolver::solve] The yinitial is empty\n";
102  throw_error(errormessage);
103  }
104 
105  // Are we integrating forward or backward?
106  double sign = xarr[1] > xarr[0] ? 1.0 : -1.0;
107 
108  // Set up the ODE system
109  gsl_odeiv2_system ode_system = { ode_equation, jacobian,
110  size_t(yinitial.size()), parameters };
111  gsl_odeiv2_driver* ode_driver = gsl_odeiv2_driver_alloc_y_new(
112  &ode_system, stepper, std::abs(hstart) * sign, abserr, relerr);
113 
114  // Initialize with the initial condition
115  double x = xarr[0];
116  DVector y(yinitial);
117  DVector dydx(nequations);
118  ode_equation(x, y.data(), dydx.data(), parameters);
119 
120  // Allocate memory for the the results: data[i][j] = y_j(x_i)
121  data = std::vector<DVector>(xarr.size(), yinitial);
122  derivative_data = std::vector<DVector>(xarr.size(), dydx);
123 
124  if (verbose) {
125  std::cout << "ODESolver step " << std::setw(5) << 0 << " / "
126  << num_x_points - 1 << " x: [" << std::setw(10) << x << "] ";
127  std::cout << "y: [";
128  for (auto& ycomp : y) {
129  std::cout << " " << std::setw(10) << ycomp << " ";
130  }
131  std::cout << "]" << std::endl;
132  }
133 
134  // Solve it step by step...
135  for (int i = 1; i < num_x_points; i++) {
136  const double xnew = xarr[i];
137 
138  if (verbose) {
139  std::cout << "ODESolver step " << std::setw(5) << i << " / "
140  << num_x_points - 1 << " x: [" << std::setw(10) << xnew << "] ";
141  }
142 
143  // Integrate from x -> xnew
144  const int status = gsl_odeiv2_driver_apply(ode_driver, &x, xnew, y.data());
145  if (status != GSL_SUCCESS) {
146  std::string errormessage = "[ODESolver::solve] GSL gsl_odeiv2_driver_apply returned non-successful\n";
147  throw_error(errormessage);
148  }
149 
150  if (verbose) {
151  std::cout << "y: [";
152  for (auto& ycomp : y) {
153  std::cout << " " << std::setw(10) << ycomp << " ";
154  }
155  std::cout << "]" << std::endl;
156  }
157 
158  // Store the derivative
159  ode_equation(x, y.data(), dydx.data(), parameters);
160 
161  // Copy over result
162  data[i] = y;
163  derivative_data[i] = dydx;
