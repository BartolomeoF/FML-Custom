35  /// //
36  /// A general multigrid solver to solve //
37  /// PDEs on a compact domain in any dimension //
38  /// with periodic or Dirichlet boundary //
39  /// conditions. //
40  /// //
41  /// The main implemention is for periodic //
42  /// boundary conditions. However the user has //
43  /// an option of providing a mask-field //
44  /// which specifies the boundary and the //
45  /// value of the field is held fixed to that //
46  /// of the initial conditions when solving //
47  /// Use set_mask to provide the mask. This is //
48  /// a grid with +1 if the cell is active and //
49  /// -1 if its a boundary cell and the boundary//
50  /// value is that provided in the initial //
51  /// guess. //
52  /// Currently this is a quite simplistic //
53  /// implemention so on a general domain //
54  /// make sure you use a big enough mask so //
55  /// that the lowest level is masked out. //
56  /// To try this compile with USE_MASK //
57  /// //
58  /// User needs to provide the equation in the //
59  /// form L(y) = 0 and dL/dy //
60  /// and the convergence criterion (we provide //
61  /// two common ones as fiducial options) //
62  /// For defining the equation one can use //
63  /// some helper functions get_Field, //
64  /// get_Gradient, get_Laplacian etc. //
65  /// Currently the list of cells provided //
66  /// only contains the closest 2*NDIM cells //
67  /// so one will have to compute by hand index //
68  /// of other cells if needed (and make sure //
69  /// that n_extra_slices are set right so that //
70  /// the cells from neighboring cpus exists) //
71  /// //
72  /// _PERIODIC is the box periodic or not? //
73  /// The standard is periodic //

239  // Poisson equation with source s.t. the solution is f(x_1,x_2,...) = Sum_i sin(2 pi x_i)
240  MultiGridFunction<NDIM,T> TestEquation = [](MultiGridSolver<NDIM,T> *sol, int level, IndexInt index) {
241  // Compute list of cell-index of closest 2NDIM+1 cells (0 is index)
242  auto index_list = sol->get_neighbor_gridindex(level, index);
243  // Compute the laplacian [D^2 f]
244  auto laplacian = sol->get_Laplacian(level, index_list);
245  // Compute the deriative of laplacian [d/df D^2 f]
246  auto derivlaplacian = sol->get_derivLaplacian(level, index_list);
247  // The right hand side of the PDE
248  auto coordinate = sol->get_Coordinate(level, index);
249  T source = 0.0;
250  for(int idim = 0; idim < NDIM; idim++)
251  source += (-4.0 * M_PI * M_PI) * std::sin(2.0 * M_PI * coordinate[idim]);
252  // The full equation L(f) = 0
253  auto l = laplacian - source;
254  // The derivtive dL/df
255  auto dl = derivlaplacian;

784  template<int NDIM, class T>
785  void MultiGridSolver<NDIM,T>::prolonge_up_array(int to_level, MPIGrid<NDIM,T>& Bottom, MPIGrid<NDIM,T>& Top){
786  int twotondim = power(2, NDIM);
787  int NTop = get_N(to_level);
788  int NBottom = NTop/2;
789 
790  IndexInt NtotLocalTop = Top.get_NtotLocal();
791 
792  // Compute NTop, Ntop^2, ... , Ntop^{Ndim-1} and similar for Nbottom
793  std::vector<IndexInt> nBottomPow(NDIM, 1);
794  for(int idim = NDIM-2; idim >= 0; idim--){
795  nBottomPow[idim] = nBottomPow[idim+1] * NBottom;
796  }
797 
798  //IndexInt ixStartLocalBottom = Bottom.get_xStartLocal();
799  //IndexInt ixStartLocalTop = Top.get_xStartLocal();
800 
801  // Trilinear prolongation
802 #ifdef USE_OMP
803 #pragma omp parallel for
804 #endif
805  for (IndexInt i = 0; i < NtotLocalTop; i++) {
806 #ifdef USE_MASK
807  if(_bmask[to_level][i] <= 0.0) continue;
808 #endif
809  std::vector<double> fac(NDIM, 0.0);
810  std::vector<int> iplus(NDIM, 0);
811 
812  // Global coordinate of top and bottom cell
813  auto coord_top = Top.globalcoord_from_index(i);
814  auto coord_bottom = coord_top;
815  for(int idim = NDIM-1; idim >= 0; idim--)
816  coord_bottom[idim] /= 2;
817 
818  // Index of bottom cell
819  IndexInt iBottom = Bottom.index_from_globalcoord(coord_bottom);
820 
821  // Compute weights
822  double norm = 1.0;
823  for(int idim = NDIM-1; idim >= 0; idim--){
824  fac[idim] = coord_top[idim] % 2 == 0 ? 0.0 : 1.0;
825  iplus[idim] = 1;
826  if(_periodic and not (idim == 0 and FML::NTasks > 1)){
827  iplus[idim] = (coord_bottom[idim] + 1 < NBottom ? 1 : 1 - NBottom);
828  }
829  iplus[idim] *= nBottomPow[idim];
830  norm *= (1.0 + fac[idim]);
831  }
832  norm = 1.0 / norm;
833 
834  //===================================================================================
835  // Do N-linear interpolation
836  // Compute the sum Top[i] = Sum fac_i * Top[iBottom + d_i]
837  // + Sum fac_i fac_j * Top[iBottom + d_i + d_j]
838  // + Sum fac_i fac_j fac_k * Top[iBottom + d_i + d_j + d_k]
839  // + ... +
840  // + fac_1 ... fac_NDIM * Top[iBottom + d_1 + ... + d_NDIM]
841  //===================================================================================
842 
843  // This routine must probably be modified for having a mask
844  T val = Bottom[iBottom];
845  for(int k = 1; k < twotondim; k++){
846  double termfac = 1.0;
847  IndexInt iAdd = 0;
848  std::bitset<NDIM> bits = std::bitset<NDIM>(k);
849  for(int j = 0; j < NDIM; j++){
850  iAdd = bits[j] * iplus[j];
851  termfac *= 1.0 + bits[j] * (fac[j] - 1.0) ;
852  }
853  val += T(termfac) * Bottom[iBottom + iAdd];
854  }
855  Top[i] = val * T(norm);
856  }
857  }
