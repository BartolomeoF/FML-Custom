495  y[index_theta_p_start + ell] = -ell / (2.0 * ell + 1) * ckoverHpdtaudx * y[index_theta_p_start + ell - 1];
496  }
497  }

499  //=============================================================================
500  // SET: Nu multipoles
501  //=============================================================================
502  for (int ell = 0; ell < n_ell_nu; ell++) {
503  if (ell < n_ell_nu_tc) {
504  y[index_nu_start + ell] = y_tight_coupling[index_nu_start_tc + ell];
505  } else {
506  y[index_nu_start + ell] = ckoverHp / (2.0 * ell + 1) * y[index_nu_start + ell - 1];
507  }
508  }
509 
510  return y;

810  void Perturbations::compute_source_functions()
811  {
812  //Utils::StartTiming("PERT::making source");
813 
814  // The x and k arrays to use to make the spline
815  // For simplicity we jusy use the same array as used when integrating the perturbations
816  auto k_array = FML::MATH::linspace(log(k_min), log(k_max), n_k_total);
817  for(auto &k : k_array) k = std::exp(k);
818  auto x_array = FML::MATH::linspace(x_start, x_end, n_x_total);
819 
820  // Compute source functions
821  DVector ST_array(n_k_total * n_x_total);
822  DVector SE_array(n_k_total * n_x_total);
823  DVector SN_array(n_k_total * n_x_total);
824 
825  DVector SW_array(n_k_total * n_x_total);
826  DVector ISW_array(n_k_total * n_x_total);
827  DVector VEL_array(n_k_total * n_x_total);
828  DVector POL_array(n_k_total * n_x_total);
829 
830  // x dependent quantities needed below
831  DVector exp_tau_array(n_x_total), chi_array(n_x_total);
832  DVector Hp_array(n_x_total), dHpdxHp_array(n_x_total), ddHpddxHp_array(n_x_total);
833  DVector g_array(n_x_total), dgdx_array(n_x_total), ddgddx_array(n_x_total);
834  for (int ix = 0; ix < n_x_total; ix++){
835  const double x = x_array[ix];
836  Hp_array[ix] = cosmo->Hp_of_x(x);
837  dHpdxHp_array[ix] = cosmo->dHpdx_of_x(x) / Hp_array[ix];
838  ddHpddxHp_array[ix] = cosmo->ddHpddx_of_x(x) / Hp_array[ix];
839  exp_tau_array[ix] = exp(-rec->tau_of_x(x));
840  chi_array[ix] = cosmo->eta_of_x(x_end) - cosmo->eta_of_x(x);
841  g_array[ix] = rec->g_tilde_of_x(x);
842  dgdx_array[ix] = rec->dgdx_tilde_of_x(x);
843  ddgddx_array[ix] = rec->ddgddx_tilde_of_x(x);
844  }
845 
846  for (int ix = 0; ix < n_x_total; ix++) {
847  const double x = x_array[ix];
848  for (int ik = 0; ik < n_k_total; ik++) {
849  const double k = k_array[ik];
850  const int index = ix + n_x_total * ik;
851 
852  const double Hp = Hp_array[ix];
853  const double dHpdxHp = dHpdxHp_array[ix];
854  const double ddHpddxHp = ddHpddxHp_array[ix];
855  const double chi = chi_array[ix];
856  const double exp_tau = exp_tau_array[ix];
857  const double g = g_array[ix];
858  const double dgdx = dgdx_array[ix];
859  const double ddgddx = ddgddx_array[ix];
860  const double ckoverHp = Constants.c * k / Hp;
861 
862  const double Theta0 = get_Theta(x, k, 0);
863  const double Pi = get_Pi(x, k);
864  //const double dPidx = dPidx_spline(x, k);
865  //const double ddPiddx = dPidx_spline.deriv_x(x, k);
866  const double dPidx = Pi_spline.deriv_x(x, k);
867  const double ddPiddx = Pi_spline.deriv_xx(x, k);
868  const double dPhidx = Phi_spline.deriv_x(x, k);
869  const double dPsidx = Psi_spline.deriv_x(x, k);
870  const double Psi = get_Psi(x, k);
871  const double v_b = get_v_b(x, k);
872  const double dv_bdx = v_b_spline.deriv_x(x, k);
873 
874  // The temperature source
875  const double SW_term = g * (Theta0 + Psi + Pi / 4.0);
876  const double ISW_term = exp_tau * (dPsidx - dPhidx);
877  const double velocity_term = -(1.0 / ckoverHp) * (dHpdxHp * g * v_b + dgdx * v_b + g * dv_bdx);
878  const double polarization_term = 3.0 / (4.0 * ckoverHp * ckoverHp) * (
879  (ddgddx * Pi + 2.0 * dgdx * dPidx + g * ddPiddx)
880  + 3.0 * dHpdxHp * (dgdx * Pi + g * dPidx)
881  + g * Pi * (ddHpddxHp + dHpdxHp*dHpdxHp));
882 
883  ST_array[index] = SW_term + ISW_term + velocity_term + polarization_term;
884  SW_array[index] = SW_term;
885  ISW_array[index] = ISW_term;
886  VEL_array[index] = velocity_term;
887  POL_array[index] = polarization_term;
888 
889  // Polarization source
890  SE_array[index] = x > -0.001 ? 0.0 : 3.0 * g * Pi / 4.0 / pow(ckoverHp * Hp * chi / Constants.c, 2);
891 
892  // Neutrino source
893  SN_array[index] = (dPsidx - dPhidx);
894  }
895  }
896 
897  // Spline up source functions
898  ST_spline.create(x_array, k_array, ST_array, "Source_Temp_x_k");
899  SE_spline.create(x_array, k_array, SE_array, "Source_Epol_x_k");
900  SN_spline.create(x_array, k_array, SN_array, "Source_Nu_x_k");
901 
902  // Individual contributions to the temperature source function
903  SW_spline.create (x_array, k_array, SW_array, "Source_SW_x_k");
904  ISW_spline.create(x_array, k_array, ISW_array, "Source_ISW_x_k");
905  VEL_spline.create(x_array, k_array, VEL_array, "Source_VEL_x_k");
906  POL_spline.create(x_array, k_array, POL_array, "Source_POL_x_k");
907 
908  //Utils::EndTiming("PERT::making source");
909  }

1048  //tcb::span<const double> Theta (y + psinfo_tight_coupling.index_theta_start, n_ell_theta);

1052  double& ddelta_bdx = dydx[psinfo_tight_coupling.index_delta_b];

1054  double& dv_bdx = dydx[psinfo_tight_coupling.index_v_b];

1056  double *dThetadx = &dydx[psinfo_tight_coupling.index_theta_start];
1057  double *dNudx = &dydx[psinfo_tight_coupling.index_nu_start];
1058  //tcb::span<double> dThetadx (dydx + psinfo_tight_coupling.index_theta_start, n_ell_theta);
1059  //tcb::span<double> dNudx (dydx + psinfo_tight_coupling.index_nu_start, n_ell_nu);
1060  //=============================================================================
1061 
1062  //=============================================================================
1063  // Cosmological parameters and variables
1064  //=============================================================================
1065  const double OmegaB = cosmo->get_OmegaB();
1066  const double OmegaCDM = cosmo->get_OmegaCDM();
1067  const double OmegaR = cosmo->get_OmegaR();
1068  const double OmegaNu = cosmo->get_OmegaNu();
1069  const double OmegaK = cosmo->get_OmegaK();
1070  const double H0 = cosmo->get_H0();
1071  const double Hp = cosmo->Hp_of_x(x);
1072  const double dHpdx = cosmo->dHpdx_of_x(x);
1073  const double eta = cosmo->eta_of_x(x);
1074  const double etaHp = eta * Hp / Constants.c;
1075  const double a = exp(x);
1076 
1077  //=============================================================================
1078  // Recombination variables
1079  //=============================================================================
1080  const double dtaudx = rec->dtaudx_of_x(x);
1081  const double ddtauddx = rec->ddtauddx_of_x(x);
1082  const double ckoverHp = Constants.c * k / Hp;
1083  const double ckoverH0 = Constants.c * k / H0;
1084  const double ckoverHpdtaudx = OmegaB > 0.0 ? ckoverHp / dtaudx : 0.0;
1085  const double theta2fac = polarization ? 8.0 / 15.0 : 20.0 / 45.0;
1086 
1087  //=============================================================================
1088  // Modifications for curvature and vector/tensor modes
1089  //=============================================================================
1090  DVector kappa ( std::max(n_ell_nu, n_ell_theta) + 1, 1.0);
1091  DVector kappa2( std::max(n_ell_nu, n_ell_theta) + 1, 1.0);
1092  const double m_type = 0;
1093  const double K = cosmo->get_K();
1094  const double qoverk = sqrt(1.0 + K/(k*k));
1095  const double q = k * qoverk;
1096  const double q2 = q*q;
1097  const double cqoverHp = ckoverHp * qoverk;
1098  for(size_t ell = 2; ell < kappa.size(); ell++){
1099  kappa[ell] = kappa_factor(ell*ell, m_type*m_type, 0.0*0.0, q2, K);
1100  kappa2[ell] = kappa_factor(ell*ell, m_type*m_type, 2.0*2.0, q2, K);
1101  }
1102 
1103  //=============================================================================
1104  // The lowest multipoles
1105  // We might or might not have all of them (and it differs with and without polarization)
1106  //=============================================================================
1107  const double Nu0 = neutrinos ? Nu[0] : 0.0;
1108  const double Nu2 = neutrinos ? Nu[2] : 0.0;
1109  const double Theta2 = -theta2fac * ckoverHpdtaudx * Theta[1];
1110  const double Theta_p0 = 5.0 / 4.0 * Theta2;
1111  const double Theta_p2 = Theta2 / 4.0;
1112  const double Pi = Theta_p0 + Theta_p2 + Theta2;
1113 
1114  //=============================================================================
1115 
1116  // The second gravitational potential
1117  const double Psi = -Phi - 12.0 * pow(1.0 / (ckoverH0 * a), 2) * (OmegaR * Theta2 + OmegaNu * Nu2);
1118 
1119  //=============================================================================
1120  // SET: Derivative of Phi
1121  //=============================================================================
1122  const double ckoverHp2_curvature = (ckoverHp * ckoverHp + 3.0 * OmegaK * (H0/Hp)*(H0/Hp));
1123  dPhidx = Psi - ckoverHp2_curvature / 3.0 * Phi + 0.5 * (H0 * H0) / (Hp * Hp * a * a) *
1124  (OmegaCDM * delta_cdm * a + OmegaB * delta_b * a + 4.0 * (OmegaR * Theta[0] + OmegaNu * Nu0));
1125 
1126  // The tight coupling factor q_tc
1127  const double R = OmegaB > 0.0 ? 4.0 / 3.0 * OmegaR / OmegaB / a : 0.0;
1128  const double sound_speed_squared = rec->get_baryon_sound_speed_squared(x);
1129  const double dTheta0dx = -cqoverHp * Theta[1] - dPhidx;
1130  const double dTheta2dx = cqoverHp / 5.0 * (2.0 * Theta[1]) + dtaudx * (Theta2 - Pi / 10.0);
1131  const double q_tc = (-((1.0 - R) * dtaudx + (1.0 + R) * ddtauddx) * (3.0 * Theta[1] + v_b) - ckoverHp * Psi
1132  + (1.0 - dHpdx / Hp) * cqoverHp * (-Theta[0] + 2.0 * Theta2) + cqoverHp * (-dTheta0dx + dTheta2dx)) / ((1.0 + R) * dtaudx + dHpdx / Hp - 1.0);
1133  const double q_vb_tc = (-v_b - ckoverHp * sound_speed_squared * delta_b + R * (q_tc + cqoverHp * (-Theta[0] + 2.0 * Theta2))) / (1.0 + R);
1134 
1135  //=============================================================================
1136  // SET: Baryons and CDM
1137  //=============================================================================
1138  ddelta_cdmdx = ckoverHp * v_cdm - 3.0 * dPhidx;
1139  ddelta_bdx = ckoverHp * v_b - 3.0 * dPhidx;
1140  dv_cdmdx = -v_cdm - ckoverHp * Psi;
1141  dv_bdx = q_vb_tc - ckoverHp * Psi;
1142 
1143  //=============================================================================
1144  // SET: Photons multipoles (Theta_ell)
1145  //=============================================================================
1146  dThetadx[0] = -cqoverHp * Theta[1];
1147  dThetadx[1] = (q_tc - q_vb_tc) / 3.0;
1148 
1149  // Add sources
1150  dThetadx[0] += -dPhidx;
1151  dThetadx[1] += ckoverHp * Psi / 3.0;
1152 
1153  //=============================================================================
1154  // SET: Neutrinos multipoles (Nu_ell)
1155  //=============================================================================
1156  for (int ell = 0; ell < n_ell_nu; ell++) {
1157  const double prefac = cqoverHp / (2.0 * ell + 1);
1158  if (ell == 0) {
1159  dNudx[ell] = prefac * ( - (ell + 1) * Nu[ell+1] );
1160  } else if (ell < n_ell_nu - 1) {
1161  dNudx[ell] = prefac * (ell * kappa[ell] * Nu[ell - 1] - (ell + 1) * kappa[ell+1] * Nu[ell + 1]);
1162  } else {
1163  dNudx[ell] = cqoverHp * kappa[ell] * Nu[ell - 1] - (ell + 1) / (etaHp) * kappa[ell+1] * Nu[ell];
1164  }
1165 
1166  // Add sources
1167  if(ell == 0) dNudx[ell] += - dPhidx;
1168  if(ell == 1) dNudx[ell] += ckoverHp / 3.0 * Psi;
1169  }
1170 
1171  return GSL_SUCCESS;
1172  }

1175  double k,

1177  double* dfdy,

1180  //Utils::StartTiming("PERT::jacobian");

1182  // This computes dfdt - explicit x-derivative of the rhs of the ODE system
1183  // and dfdy the Jacobian matrix of the system
1184  const double& delta_cdm = y[psinfo.index_delta_cdm];
1185  const double& delta_b = y[psinfo.index_delta_b];
1186  const double& v_cdm = y[psinfo.index_v_cdm];

1188  const double& Phi = y[psinfo.index_Phi];

1190  const double* Theta_p = &y[psinfo.index_theta_p_start];
1191  const double* Nu = &y[psinfo.index_nu_start];
1192 
1193  double& dfdt_delta_cdm = dfdt[psinfo.index_delta_cdm];
1194  double& dfdt_delta_b = dfdt[psinfo.index_delta_b];
1195  double& dfdt_v_cdm = dfdt[psinfo.index_v_cdm];
1196  double& dfdt_v_b = dfdt[psinfo.index_v_b];
1197  double& dfdt_Phi = dfdt[psinfo.index_Phi];
1198  double* dfdt_Theta = &dfdt[psinfo.index_theta_start];
1199  double* dfdt_Theta_p = &dfdt[psinfo.index_theta_p_start];
1200  double* dfdt_Nu = &dfdt[psinfo.index_nu_start];
1201 
1202  // Cosmology variables
1203  const double OmegaB = cosmo->get_OmegaB();
1204  const double OmegaCDM = cosmo->get_OmegaCDM();
1205  const double OmegaR = cosmo->get_OmegaR();
1206  const double OmegaNu = cosmo->get_OmegaNu();
1207  const double H0 = cosmo->get_H0();
1208  const double Hp = cosmo->Hp_of_x(x);
1209  const double dlogHpdx = cosmo->dHpdx_of_x(x) / Hp;
1210  const double eta = cosmo->eta_of_x(x);
1211  const double etaHp = eta * Hp / Constants.c;
1212  const double a = exp(x);
1213  const double doneoveretaHpdx = -1.0 / (etaHp) * (1.0 / etaHp + dlogHpdx);
1214 
1215  // Recombination variables
1216  const double dtaudx = rec->dtaudx_of_x(x);
1217  const double ddtauddx = rec->ddtauddx_of_x(x);
1218  const double ckoverHp = Constants.c * k / Hp;
1219  const double ckoverH0 = Constants.c * k / H0;
1220 
1221  // Some quantitiess needed below
1222  const double Nu0 = psinfo.n_ell_nu > 0 ? Nu[0] : 0.0;
1223  const double Nu2 = psinfo.n_ell_nu > 2 ? Nu[2] : 0.0;
1224  const double Theta_p0 = psinfo.n_ell_theta_p > 0 ? Theta_p[0] : 0.0;
1225  const double Theta_p2 = psinfo.n_ell_theta_p > 2 ? Theta_p[2] : 0.0;
1226  const double PhiH0H0term = 0.5 * (H0 * H0) / (Hp * Hp * a * a);
1227  const double Psiterm = 12.0 / (ckoverH0 * ckoverH0 * a * a);
1228  const double R = OmegaB > 0.0 ? 4.0 / 3.0 * OmegaR / OmegaB / a : 0.0;
1229  const double Pi = Theta[2] + Theta_p0 + Theta_p2;
1230  const double Psi = -Phi - Psiterm * (OmegaR * Theta[2] + OmegaNu * Nu2);
1231  const double dPsidt = 2.0 * Psiterm * (OmegaR * Theta[2] + OmegaNu * Nu2);
1232  ;
1233 
1234  // Derivatives of scalar equations
1235  dfdt_Phi = dPsidt + 2.0 * dlogHpdx * ckoverHp * ckoverHp * Phi / 3.0 + PhiH0H0term
1236  * (+(-1.0 - 2 * dlogHpdx) * (OmegaCDM * delta_cdm * a + OmegaB * delta_b * a) + (-2.0 - 2 * dlogHpdx) * 4.0 * (OmegaR * Theta[0] + OmegaNu * Nu0));
1237  dfdt_delta_cdm = -dlogHpdx * ckoverHp * v_cdm - 3.0 * dfdt_Phi;
1238  dfdt_delta_b = -dlogHpdx * ckoverHp * v_b - 3.0 * dfdt_Phi;
1239  dfdt_v_cdm = dlogHpdx * ckoverHp * Psi - ckoverHp * dPsidt;
1240  dfdt_v_b = dlogHpdx * ckoverHp * Psi - ckoverHp * dPsidt + (ddtauddx - dtaudx) * R * (Theta[1] + v_b);
1241 
1242  // Photons
1243  for (int ell = 0; ell < psinfo.n_ell_theta; ell++) {
1244  if (ell == 0) {
1245  dfdt_Theta[0] = dlogHpdx * ckoverHp * Theta[1] - dfdt_Phi;
1246  } else if (ell == 1) {
1247  dfdt_Theta[1] = -dlogHpdx * ckoverHp / (2.0 * ell + 1) * (ell * Theta[0] - (ell + 1) * Theta[2] + Psi) + ckoverHp / 3.0 * dPsidt + ddtauddx * (Theta[ell] + v_b / 3.0);
1248  } else if (ell < psinfo.n_ell_theta - 1) {
1249  dfdt_Theta[ell] = -dlogHpdx * ckoverHp / (2.0 * ell + 1) * (ell * Theta[ell - 1] - (ell + 1) * Theta[ell + 1]) + ddtauddx * (Theta[ell] - (ell == 2 ? Pi / 10.0 : 0.0));
1250  } else {
1251  dfdt_Theta[ell] = -dlogHpdx * ckoverHp * Theta[ell - 1] - (ell + 1) * doneoveretaHpdx * Theta[ell] + ddtauddx * Theta[ell];
1252  }
1253  }
1254 
1255  // Polarization (Assuming nellp > 1)
1256  for (int ell = 0; ell < psinfo.n_ell_theta_p; ell++) {
1257  if (ell == 0) {
1258  dfdt_Theta_p[ell] = dlogHpdx * ckoverHp * Theta_p[1] + ddtauddx * (Theta_p[ell] - Pi / 2.0);
1259  } else if (ell < psinfo.n_ell_theta_p - 1) {
1260  dfdt_Theta_p[ell] = -dlogHpdx * ckoverHp / (2.0 * ell + 1) * (ell * Theta_p[ell - 1] - (ell + 1) * Theta_p[ell + 1]) + ddtauddx * (Theta_p[ell] - (ell == 2 ? Pi / 10.0 : 0.0));
1261  } else {
1262  dfdt_Theta_p[ell] = -dlogHpdx * ckoverHp * Theta_p[ell - 1] - (ell + 1) * doneoveretaHpdx * Theta_p[ell] + ddtauddx * Theta_p[ell];
1263  }
1264  }
1265 
1266  // Neutrinos (Assuming nellnu > 2)
1267  for (int ell = 0; ell < psinfo.n_ell_nu; ell++) {
1268  if (ell == 0) {
1269  dfdt_Nu[ell] = dlogHpdx * ckoverHp * Nu[1] - dfdt_Phi;
1270  } else if (ell == 1) {
1271  dfdt_Nu[ell] = -dlogHpdx / 3.0 * ckoverHp * (Nu[0] - 2.0 * Nu[2] + Psi) + ckoverHp / 3.0 * dPsidt;
1272  } else if (ell < psinfo.n_ell_nu - 1) {
1273  dfdt_Nu[ell] = -dlogHpdx * ckoverHp / (2.0 * ell + 1) * (ell * Nu[ell - 1] - (ell + 1) * Nu[ell + 1]);
1274  } else {
1275  dfdt_Nu[ell] = -dlogHpdx * ckoverHp * Nu[ell - 1] - (ell + 1) * doneoveretaHpdx * Nu[ell];
1276  }
1277  }
1278

1281

1293  if (psinfo.n_ell_nu > 2) {
1294  cur[psinfo.index_nu_start + 2] = 3.0 * Psiterm * OmegaNu;

1296 
1297  // delta_b

1301  cur[psinfo.index_v_b] = ckoverHp;
1302  cur[psinfo.index_Phi] = 3.0 * (1.0 + ckoverHp * ckoverHp / 3.0);

1310  }
1311 
1312  // v_cdm
1313  cur = &jac[psinfo.index_v_cdm][0];
1314  cur[psinfo.index_v_cdm] = -1.0;
1315  cur[psinfo.index_Phi] = ckoverHp;

1318  cur[psinfo.index_nu_start + 2] = ckoverHp * Psiterm * OmegaNu;
