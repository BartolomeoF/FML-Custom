235  MPIGrid<NDIM,T>::MPIGrid(int N, bool periodic, int n_extra_slices_left, int n_extra_slices_right) {
236  assert_mpi(N > 0,
237  "[MPIGrid] We need Ngrid > 0\n");
238  assert_mpi(n_extra_slices_left >= 0 and n_extra_slices_right >= 0,
239  "[MPIGrid] Number of extra slices cannot be negative\n");
240  if(N % FML::NTasks != 0 and FML::ThisTask == 0)
241  std::cout << "[MPIGrid] Warning: FML::NTasks should divide N to be compatible with other MPI methods\n";
242 
243  // Compute slices for task
244  std::vector<int> slices_per_task(FML::NTasks, N / FML::NTasks);
245  int nmore = N % FML::NTasks;
246  int sumslicesbeforethistask = 0;
247  for(int task = 0; task < FML::NTasks; task++){
248  if(task < nmore)
249  slices_per_task[task] += 1;
250 
251  if(task < FML::ThisTask)
252  sumslicesbeforethistask += slices_per_task[task];
253  }
254 
255  _periodic = periodic;
256  _N = N;
257  _NLocal = slices_per_task[FML::ThisTask];
258  _xStartLocal = sumslicesbeforethistask;
259  _n_extra_slices_left = n_extra_slices_left;
260  _n_extra_slices_right = n_extra_slices_right;
261  _NperSlice = FML::power(_N, NDIM-1);
262  _NtotLocalLeft = _NperSlice * _n_extra_slices_left;
263  _NtotLocalRight = _NperSlice * _n_extra_slices_right;
264  _NtotLocal = _NperSlice * _NLocal;
265  _Ntot = _NperSlice * _N;
266  _NtotLocalAlloc = _NtotLocalLeft + _NtotLocal + _NtotLocalRight;
267  _RightTask = (FML::ThisTask + 1) % FML::NTasks;
268  _LeftTask = (FML::ThisTask - 1 + FML::NTasks) % FML::NTasks;
269 
270  if(_NLocal == 0){
271  // Tasks with 0 cells does nothing
272  _RightTask = _LeftTask = FML::ThisTask;
273  _n_extra_slices_left = _n_extra_slices_right = 0;
274  _NtotLocalLeft = _NtotLocalRight = _NtotLocal = _NtotLocalAlloc = 0;
275  } else {
276  // Find right CPU with cells
277  while( slices_per_task[_RightTask] == 0){
278  _RightTask = (_RightTask + 1) % FML::NTasks;
279  }
280  // Find left CPU with cells
281  while( slices_per_task[_LeftTask] == 0){
282  _LeftTask = (_LeftTask - 1 + FML::NTasks) % FML::NTasks;
283  }
284 
285  // This is maybe something we should have... but fails with multigrid
286  //assert_mpi(_n_extra_slices_right <= slices_per_task[_RightTask], "");
287  //assert_mpi(_n_extra_slices_left <= slices_per_task[_LeftTask], "");
288  }
289 
290  // Allocate memory
291  _y.resize(_NtotLocalAlloc);
292  add_memory_label("MPIGrid");
293 
294  // Show some info
295 #ifdef DEBUG
296  if(FML::ThisTask == 0){
297  std::cout << "\n=======================\nCreating Grid:\n";
298  std::cout << "FML::ThisTask: " << FML::ThisTask << " NLocal: " << _NLocal << " xStartLocal: " << _xStartLocal
299  << " LeftTask: " << _LeftTask << " RightTask: " << _RightTask << "\n" << std::flush;
300  }
301 #endif

552  template<int NDIM, class T>
553  void MPIGrid<NDIM,T>::communicate_boundaries(){
554  const int nsend_to_left = _n_extra_slices_right;
555  const int nsend_to_right = _n_extra_slices_left;
556 
557  // Buffer to recive data in
558  std::vector<T> recv_array(_NperSlice);
559 
560  // Send rightmost slices right and store in extra left slices
561  for(int i = 0; i < nsend_to_right; i++){
562  const bool do_not_store = not _periodic and FML::ThisTask == 0;
563  T *slice_left_torecv = _y.data() + _NperSlice * i;
564  send_slice_right(_NLocal - nsend_to_right + i, recv_array);
565  if(not do_not_store) memcpy(slice_left_torecv, recv_array.data(), _NperSlice * sizeof(T));
566  }
567 
568  // Send leftmost slices left and store in extra right slices
569  for(int i = 0; i < nsend_to_left; i++){
570  const bool do_not_store = not _periodic and FML::ThisTask == FML::NTasks-1;
571  T *slice_right_torecv = _y.data() + _NtotLocalLeft + _NtotLocal + _NperSlice * i;
572  send_slice_left(i, recv_array);
573  if(not do_not_store) memcpy(slice_right_torecv, recv_array.data(), _NperSlice * sizeof(T));
574  }
575  }

577  template<int NDIM, class T>
578  double MPIGrid<NDIM,T>::norm(){
579  double norm2 = 0.0;
580 #ifdef USE_OMP
581 #pragma omp parallel for reduction(+:norm2)
582 #endif
583  for(IndexInt i = 0; i < _NtotLocal; i++){
584  norm2 += AbsoluteValue(get_y(i)) * AbsoluteValue(get_y(i));
585  }
586 #ifdef USE_MPI
