33  ///
34  /// Class for holding grids and performing real-to-complex and complex-to-real
35  /// FFTs using FFTW with MPI. Templated on dimension.
36  ///
37  /// The real and fourier grid is stored in the same array to save memory
38  /// and all transforms are done in-place. Keeps track of the status of the grid, i.e.
39  /// if its in real-space or fourier-space (set with set_grid_status_real)
40  ///
41  /// The default constructor arguments in (Nmesh, n_extra_left, n_extra_right) is:
42  ///
43  /// N : Dimension of the grid
44  ///
45  /// Nmesh : Number of grid-nodes per dimension (assuming the same)
46  ///
47  /// n_extra : Alloc extra slices of the grid in the x-dimension (left and/or right)
48  ///
49  /// Compile-time defines:
50  ///
51  /// NO_AUTO_FFTW_MPI_INIT : Do not automatically initialize FFTW, handle this yourself
52  ///
53  /// BOUNDSCHECK_FFTWGRID : bound checks when setting and getting values
54  ///
55  /// SINGLE_PRECISION_FFTW : use float instead of double
56  ///
57  /// LONG_DOUBLE_PRECISION_FFTW : use load double instead of double
58  ///
59  /// DEBUG_FFTWGRID : Show some info while running
60  ///
61  /// USE_MPI : Use MPI
62  ///
63  /// USE_OMP : Use OpenMP
64  ///
65  /// USE_FFTW_THREADS : Use threads if possible. With this we assume the maximum number of threads
66  /// If you want to use fewer then you can call create_wisdom(..., nthreads) to change this
67  ///
68  /// See FFTWGlobal.h for type definitions and the defines that deal with the precision we want.
69  ///
70  /// External variables/methods we rely on:
71  ///
72  /// ThisTask : The MPI task number
73  ///
74  /// NTasks : Number of MPI tasks
75  ///
76  /// assert_mpi(Expr, Msg)
77  ///
78  /// T power(T base, int exponent);
79  ///
80  /// The grid-stucture is only compatible with in-place FFTW transforms so

578  void FFTWGrid<N>::communicate_boundaries(){
579  int n_to_recv_right = n_extra_x_slices_right;
580  int n_to_recv_left = n_extra_x_slices_left;
581  if(n_to_recv_right > Local_nx) n_to_recv_right = Local_nx;
582  if(n_to_recv_left > Local_nx) n_to_recv_left = Local_nx;
583 
584 #ifdef DEBUG_FFTWGRID
585  if(FML::ThisTask == 0){
586  std::cout << "[FFTWGrid::communicate_boundaries] Recieving " << n_to_recv_right << " from the right and " << n_to_recv_left << " slices from the left\n";
587  }
588 #endif
589 
590 #ifdef USE_MPI
591  int rightcpu = (FML::ThisTask + 1 ) % FML::NTasks;
592  int leftcpu = (FML::ThisTask - 1 + FML::NTasks) % FML::NTasks;
593 #endif
594  int bytes_slice = NmeshTotRealSlice * sizeof(FloatType);
595 
596  for(int i = 0; i < n_to_recv_right; i++){
597  FloatType *slice_left_tosend = get_real_grid() + NmeshTotRealSlice * (i);
598  FloatType *slice_right_torecv = get_real_grid_right() + NmeshTotRealSlice * (i);

600  char *recvbuf = reinterpret_cast<char*> (slice_right_torecv);
601 #ifdef USE_MPI
602  MPI_Status status;
603  MPI_Sendrecv(
604  sendbuf, bytes_slice, MPI_CHAR, leftcpu, 0,
605  recvbuf, bytes_slice, MPI_CHAR, rightcpu, 0,
606  MPI_COMM_WORLD, &status);
607 #else
608  std::memcpy(recvbuf, sendbuf, bytes_slice);
609 #endif
610  }
611 
612  for(int i = 0; i < n_to_recv_left; i++){
613  FloatType *slice_right_tosend = get_real_grid() + NmeshTotRealSlice * (Local_nx - 1 - i);
614  FloatType *slice_left_torecv = get_real_grid_left() + NmeshTotRealSlice * (n_extra_x_slices_left - 1 - i);
615  char *sendbuf = reinterpret_cast<char*> (slice_right_tosend);
616  char *recvbuf = reinterpret_cast<char*> (slice_left_torecv);
617 
618 #ifdef USE_MPI
619  MPI_Status status;
620  MPI_Sendrecv(
621  sendbuf, bytes_slice, MPI_CHAR, rightcpu, 0,
622  recvbuf, bytes_slice, MPI_CHAR, leftcpu, 0,
623  MPI_COMM_WORLD, &status);
624 #else
625  std::memcpy(recvbuf, sendbuf, bytes_slice);
626 #endif
627  }

666  "[FFTWGrid] The number of CPUs should divide the gridsize. Otherwise there might be issues with extra padding in FFTW and in they way we divide the domain\n");

670  std::vector<ptrdiff_t> NmeshPerDimFourier(N,Nmesh);

830  template<int N>
831  void FFTWGrid<N>::fftw_r2c(){
832 #ifdef USE_FFTW
833 
834 #ifdef DEBUG_FFTWGRID
835  if(FML::ThisTask == 0){
836  std::cout << "[FFTWGrid::fftw_r2c] Transforming grid to fourier space\n";
837  }
838  if(not grid_is_in_real_space){
839  if(FML::ThisTask == 0)
840  std::cout << "Warning: [FFTWGrid::fftw_r2c] Transforming grid whose status is already [Fourierspace]\n";
841  }
842 #endif
843 
844  //=================================================================================
845  // Make a copy of the first few cells that might be overwritten when doing the FFT
846  // This might mess up extra right slices we have if we don't make a copy here
847  //=================================================================================
848  FloatType *real_grid_right = get_real_grid_right();
849  std::vector<FloatType> right_copy;
850  if(n_extra_x_slices_right > 0){
851  right_copy = std::vector<FloatType>(Nmesh/2+1);
852  for(int i = 0; i < Nmesh/2+1; i++) right_copy[i] = real_grid_right[i];
853  }
854 
855 #ifdef USE_MPI
856  std::vector<ptrdiff_t> NmeshPerDim(N,Nmesh);

858 #else
859  std::vector<int> NmeshPerDim(N,Nmesh);
860  my_fftw_plan plan_r2c = MAKE_PLAN_R2C(N, NmeshPerDim.data(), get_real_grid(), get_fftw_grid(), FFTW_ESTIMATE);
861 #endif
862 
863  EXECUTE_FFT(plan_r2c);
864  grid_is_in_real_space = false;
865 
866  // Normalize
867  double norm = 1.0 / std::pow(double(Nmesh), N);
868  auto *fourier_grid = get_fourier_grid();
869  for(auto && index: get_fourier_range()){
870  fourier_grid[index] *= norm;
871  }
872 
873  //=================================================================================
874  // Copy back data we copied
875  //=================================================================================
876  if(n_extra_x_slices_right > 0){
877  for(int i = 0; i < Nmesh/2+1; i++) real_grid_right[i] = right_copy[i];
878  }
879 
880  DESTROY_PLAN(plan_r2c);
881 #else
882  assert_mpi(false,
883  "[FFTWGrid::fftw_r2c] Compiled without FFTW support so cannot take Fourier transforms\n");
884 #endif
