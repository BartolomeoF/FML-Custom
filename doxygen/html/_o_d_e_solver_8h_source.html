17 //
18 // This is a wrapper around the GSL library to easily
19 // solve ODEs and return the data in whatever format
20 // you want. Safe to use within OpenMP threads as
21 // long as spline is not created within a thread!
22 //
23 // Supplying an x-array the solution will be stored at
24 // each of the points in the array (must be monotonic)
25 //
26 // Example use:
27 // Solve the system dy0/dx=y1, dy1/dx=-y0 with
28 // y0 = 2.0 and y1 = -2.0 on the interval [0,1]
29 // and store the solution at x = 0.0, 0.25, 0.5 and 1.0
30 //---------------------------------------------------
31 // ODEFunction dydx = [&](double x, const double *y, double *dydx){
32 // dydx[0] = y[1];
33 // dydx[1] = -y[0];
34 // return GSL_SUCCESS;
35 // };
36 // DVector yini{2.0, -2.0}
37 // DVector x_array{0.0, 0.25, 0.5, 1.0};
38 // ODESolver ode;
39 // ode.solve(dydx, x_array, yini);
40 // auto solution = ode.get_data();
