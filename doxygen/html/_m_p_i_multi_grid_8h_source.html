194  MPIMultiGrid<NDIM,T>::MPIMultiGrid(
195  int N,
196  int Nlevel,
197  bool periodic,
198  int n_extra_slices_left,
199  int n_extra_slices_right) :
200  _N(N),
201  _periodic(periodic),
202  _n_extra_slices_left(n_extra_slices_left),
203  _n_extra_slices_right(n_extra_slices_right)
204  {
205 
206  //==================================================================================
207  // Check that FML::NTasks is a power of 2 if we have more than 1 task
208  // If its not then the parallelization becomes much harder
209  // Also check that N is a power of 2
210  //==================================================================================
211  assert_mpi( N > 0 and FML::power(2, intlog2(N)) == N ,
212  "[MPIMultiGrid] N must be positive and a power of 2 for this class too work\n");
213  if(FML::NTasks > 1){
214  assert_mpi( FML::power(2, intlog2(FML::NTasks)) == FML::NTasks,
215  "[MPIMultiGrid] FML::NTasks must be a power of 2 for this class too work with MPI\n");
216  }
217 
218  //==================================================================================
219  // The smallest level we allow is one where we have one cell per CPU on the coarsest level
220  // just makes the algorithms easier as not all tasks will have cells and right/left tasks
221  // might lie several CPUs apart
222  //==================================================================================
223  if(Nlevel < 0){
224  if(FML::NTasks > 1) {
225  Nlevel = intlog2(_N) - intlog2(FML::NTasks) + 1;
226  } else {
227  Nlevel = intlog2(_N);
228  }
229  }
230  _Nlevel = Nlevel;
231  _NinLevel = std::vector<int>(_Nlevel, _N);
232  _y.resize(_Nlevel);
233 
234  // Check that _Nlevel is OK
235  assert_mpi( _Nlevel > 0 ,
236  "[MPIMultiGrid] Nlevel must be > 1 (otherwise its no multigrid)\n");
237  if(FML::NTasks > 0){
238  assert_mpi( _Nlevel <= intlog2(N) - intlog2(FML::NTasks) + 1,
239  "[MPIMultiGrid] Nlevel is too large. With MPI the smallest level corresponds to each task having jus 2 cell each\n");
240  } else {
241  assert_mpi( _Nlevel <= intlog2(_N),
242  "[MPIMultiGrid] Nlevel is too large. Coarsest level will have less than 1 cell\n");
243  }
244 
245  //==================================================================================
246  // Make all grids
247  //==================================================================================
248  _y[0] = MPIGrid<NDIM, T> (_N, _periodic, _n_extra_slices_left, _n_extra_slices_right);
249  _NtotLocal = _y[0].get_NtotLocal();
250  _NtotLocalinLevel.resize(_Nlevel);
251  _NtotLocalinLevel[0] = _NtotLocal;
252  _NinLevel.resize(_Nlevel);
253  _NinLevel[0] = _N;
254  for(int level = 1; level < _Nlevel; level++){
255  _NinLevel[level] = _NinLevel[level-1] / 2;
256  _y[level] = MPIGrid<NDIM, T>(_NinLevel[level], _periodic, _n_extra_slices_left, _n_extra_slices_right);
257  _NtotLocalinLevel[level] = _y[level].get_NtotLocal();
258  }
259  }

262  void MPIMultiGrid<NDIM,T>::restrict_down(int from_level, MPIGrid<NDIM,T> &to_grid){
263  // Cannot restict down if we are at the bottom
264  if(from_level + 1 >= _Nlevel) return;
265 
266  // Sanity check
267  assert_mpi( to_grid.get_N() == _y[from_level + 1].get_N() ,
268  "[MPIMultiGrid::restrict_down] Grid we restrict down to has wrong size\n");
269 
270  // One over number of cells averaged over [ = 1 / 2^Ndim ]
271  T oneovernumcells = T ( 1.0/double(FML::power(2,NDIM)) );
272 
273  // Pointers to Top and Bottom grid
274  auto& TopGrid = _y[from_level];
275  auto& BottomGrid = to_grid;
276 
277  // Clear bottom array
278  IndexInt NtotLocalBottom = BottomGrid.get_NtotLocal();
279  std::fill_n(&BottomGrid[0], NtotLocalBottom, T(0.0));
280 
281  // Loop over top grid
282  IndexInt NtotLocalTop = TopGrid.get_NtotLocal();
283  for (IndexInt index_top = 0; index_top < NtotLocalTop; index_top++) {
284 
285  // Convert global coordinate of current cell in top to bottom
286  auto coord = TopGrid.globalcoord_from_index(index_top);
287  for(int idim = 0; idim < NDIM; idim++){
288  coord[idim] = coord[idim]/2;
289  }
290 
291  // Add up to restricted grid
292  IndexInt index_bottom = BottomGrid.index_from_globalcoord(coord);
293  BottomGrid[index_bottom] += TopGrid[index_top] * oneovernumcells;
294  }
295  }
