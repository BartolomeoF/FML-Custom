107  void add(T & particle, bool periodic){
108  double _mass = particle.get_mass();
109  auto *_pos = particle.get_pos();
110  auto *_vel = particle.get_vel();
111 
112  // Initialize
113  if(np == 0){
114  for(int idim = 0; idim < NDIM; idim++){
115  pos[idim] = 0.0;
116  vel[idim] = 0.0;
117  }
118  mass = 0.0;
119  vel2 = 0.0;
120  }
121 
122  // Update center of mass
123  double dx[NDIM], v2 = 0;
124  for(int idim = 0; idim < NDIM; idim++){
125  dx[idim] = _pos[idim] - pos[idim];
126  if(periodic){
127  if(dx[idim] < -0.5) dx[idim] += 1;
128  if(dx[idim] >= 0.5) dx[idim] -= 1;
129  }
130  pos[idim] += dx[idim] * _mass / (mass + _mass);
131  if(periodic){
132  if(pos[idim] < 0.0) pos[idim] += 1;
133  if(pos[idim] >= 1.0) pos[idim] -= 1;
134  }
135  // Add velocity if particle has velocity
136  if(_vel != nullptr){
137  vel[idim] = (vel[idim] * mass + _vel[idim] * _mass) / (mass + _mass);
138  v2 += vel[idim]*vel[idim];
139  }
140  }
141  // Update <v^2>
142  vel2 = (vel2 * mass + _mass * v2) /(mass + _mass);
143  np++;
144  mass += _mass;
145  }

147  // Merge the groups
148  void merge(struct FoFHalo &g, bool periodic){
149  if(g.np == 0) return;
150  if(np == 0){
151  assert(false); // Should not happen
152  }
153 
154  // Merge the two centers of mass
155  double dx[NDIM];
156  for(int idim = 0; idim < NDIM; idim++){
157  dx[idim] = g.pos[idim] - pos[idim];
158  if(periodic){
159  if(dx[idim] < -0.5) dx[idim] += 1;
160  if(dx[idim] >= 0.5) dx[idim] -= 1;
161  }
162  // Update center of mass
163  pos[idim] += dx[idim] * g.mass/double(mass+g.mass);
164  if(periodic){
165  if(pos[idim] < 0.0) pos[idim] += 1;
166  if(pos[idim] >= 1.0) pos[idim] -= 1;
167  }
168  // Update COM velocity
169  vel[idim] = (vel[idim] * mass + g.vel[idim] * g.mass) / (mass+g.mass);
170  }
171  // Update <v^2>
172  vel2 = (vel2 * mass + g.vel2 * g.mass) /(mass + g.mass);
173  np += g.np;
174  g.np = 0;
175  }

577  MPI_Send(BoundaryDataToSendBack_pindex.data(), bytes_send2, MPI_BYTE, LeftTask, 0, MPI_COMM_WORLD);
578  if(FML::ThisTask == sendTask)
579  MPI_Recv(BoundaryParticleIndex.data(), bytes_recv2, MPI_BYTE, RightTask, 0, MPI_COMM_WORLD, &status);
580 
581  if(FML::ThisTask == recvTask)
582  MPI_Send(BoundaryDataToSendBack_fofindex.data(), bytes_send2, MPI_BYTE, LeftTask, 0, MPI_COMM_WORLD);
583  if(FML::ThisTask == sendTask)
584  MPI_Recv(BoundaryParticleRightFoFIndex.data(), bytes_recv2, MPI_BYTE, RightTask, 0, MPI_COMM_WORLD, &status);
585  }
586 
587  // Update FoF group indices
588  if(FML::ThisTask == sendTask or merging_in_parallel){
589 
590  for(size_t i = 0; i < count_recv; i++){
591  const size_t pindex = BoundaryParticleIndex[i];
592  assert(pindex < NumPart);
593  const size_t NewFoFIndex = BoundaryParticleRightFoFIndex[i];
594  assert(NewFoFIndex != no_FoF_ID);
595  const size_t OldFoFIndex = particle_id_FoF[pindex];
596  if(OldFoFIndex == NewFoFIndex) continue;
597  if(OldFoFIndex == no_FoF_ID){
598  particle_id_FoF[pindex] = NewFoFIndex;
599  } else {
600 #ifdef USE_OMP
601 #pragma omp parallel for
602 #endif
603  for(size_t j = 0; j < NumPart; j++)
604  if(particle_id_FoF[j] == OldFoFIndex){
605  particle_id_FoF[j] = NewFoFIndex;
606  }
607  }
608  }
609  }
610 
611  // We now finally have the list of particles with links
612 #ifdef DEBUG_FOF
613  if(FML::ThisTask == sendTask or merging_in_parallel)
614  std::cout << FML::ThisTask << " has " << nboundary_right << " boundary particles and " << count_recv << " of them have links" << std::endl;
615 #endif
616  }
617  if(!merging_in_parallel) MPI_Barrier(MPI_COMM_WORLD);
618  }
619 #endif
620  }

622  template<class T, int NDIM>
623  void FriendsOfFriendsLinkingLocal(
624  T *part,
625  size_t NumPart,
626  std::vector<FoFCells> &PartCells,
627  int Ngrid,
628  int Local_nx,
629  double fof_distance,
630  std::vector<size_t> & particle_id_FoF,
631  bool periodic)
632  {
633 
634  //=========================================================================
635  // Link together all particles in the local domain that are located within
636  // a linkking distance of linking_length times the mean particle separation.
637  // The result is a list of all the particles and an associated FoF group ID
638  // This ID can be arbitrary but is guaranteed to be unique across tasks
639  //=========================================================================
640 
641  const double fof_distance2 = fof_distance * fof_distance;
642  const size_t NgridTot = size_t(Local_nx) * FML::power(Ngrid,NDIM-1);
643  const constexpr int nblocksearchpartgrid = 3;
644  const int threetondim = FML::power(nblocksearchpartgrid,NDIM);
645 
646  // Some basic checks
647  assert(part[0].get_ndim() == NDIM);
648 
649 #ifdef DEBUG_FOF
650  if(FML::ThisTask == 0){
651  std::cout << "FriendsOfFriendsLinkingLocal\n";
652  std::cout << "FoF Linking Distance: " << fof_distance << "\n";
653  std::cout << "FoF Gridsize Ngrid = " << Ngrid << " Local: " << Local_nx << "\n";
654  }
655 #endif
656 
657  // Check that the grid is not too small
658  assert(1.0/Ngrid > fof_distance);
659 
660  // Get index of the cell the particle lives in
661  auto get_cell_index = [&](size_t ipart, std::vector<int> & coord){
662  auto *pos = part[ipart].get_pos();
663  size_t index_cell = 0;
664  for(int idim = 0; idim < NDIM; idim++){
665  coord[idim] = int((pos[idim] - FML::xmin_domain * (idim==0 ? 1 : 0)) * Ngrid);
666  index_cell = index_cell * Ngrid + coord[idim];
667  assert(coord[idim] >= 0 and coord[idim] < (idim == 0 ? Local_nx : Ngrid));
668  }
669  assert(index_cell >= 0 and index_cell < NgridTot);
670  return index_cell;
671  };
672 
673  // ID of the FoF groups
674  int nFoF = 0;
675 
676  // Allocate the FoF id for each particle
677  particle_id_FoF = std::vector<size_t>(NumPart, no_FoF_ID);
678 
679  // We now have cells with particles in them. Start linking
680  bool skip = false;
681  std::vector<int> coord(NDIM);
682  for(size_t i = 0; i < NumPart; i++){
683 
684  // Current particle
685  const auto *pos1 = part[i].get_pos();
686 
687  // Compute coord of cell particle is in
688  get_cell_index(i, coord);
689 
690  size_t id_FoF_group = particle_id_FoF[i];
691 
692  // If the current particle already have a FoF ID then we don't
693  // increase nFoF
694  if(id_FoF_group == no_FoF_ID){
695  id_FoF_group = nFoF;
696  particle_id_FoF[i] = id_FoF_group;
697  nFoF++;
698  }
699 
700  // Loop over all 3^NDIM neighbor cells (center cell is included)
701  for(int nbcell = 0; nbcell < threetondim; nbcell++){
702  int icoord[NDIM];
703  for(int idim = 0, threepow = 1; idim < NDIM; idim++, threepow *= nblocksearchpartgrid){
704  int go_left_right_of_stay = -nblocksearchpartgrid/2 + (nbcell/threepow % nblocksearchpartgrid);
705  icoord[idim] = coord[idim] + go_left_right_of_stay;
706  }
707 
708  // For boundary cells in the x direction we don't have a left or a right nbor
709  // unless we only have 1 task
710  if(FML::NTasks > 1){
711  if(icoord[0] < 0) continue;
712  if(icoord[0] >= Local_nx) continue;
713  }
714 
715  // Compute cell-index or nbor cell
716  size_t index_nbor_cell = 0;
717  if(!periodic) skip = false;
718  for(int idim = 0; idim < NDIM; idim++){
719  // Periodic boundary conditions
720  if(periodic){
721  if(icoord[idim] < 0) icoord[idim] += Ngrid;
722  if(icoord[idim] >= Ngrid) icoord[idim] -= Ngrid;
723  } else {
724  if(icoord[idim] < 0) skip = true;
725  if(icoord[idim] >= Ngrid) skip = true;
726  }
727  index_nbor_cell = index_nbor_cell * Ngrid + icoord[idim];
728  }
729  if(!periodic and skip) continue;
730 
731  // Loop over all particles in nbor cell
732  const size_t np = PartCells[index_nbor_cell].np;
733  for(size_t ii = 0; ii < np; ii++){
734  const auto pindex = PartCells[index_nbor_cell].ParticleIndex[ii];
735  if(pindex == i) continue;
736  const auto *pos2 = part[pindex].get_pos();
737 
738  // Do the linking
739  double dist2 = 0.0;
740  for(int idim = 0; idim < NDIM; idim++){
741  double dx = pos2[idim] - pos1[idim];
742  if(periodic){
743  if(dx > 0.5) dx -= 1.0;
744  if(dx < -0.5) dx += 1.0;
745  }
746  dist2 += dx*dx;
747  if(dist2 > fof_distance2) {
748  break;
749  }
750  }
751 
752  // Check if particle pair should be linked
753  if(dist2 < fof_distance2){
754 
755  const auto id_existing_FoF_group = particle_id_FoF[pindex];
756 
757  if(id_existing_FoF_group == no_FoF_ID){
758 
759  // The particle has not been touched before and is added to current FoF group
760  particle_id_FoF[pindex] = id_FoF_group;
761 
762  } else {
763 
764  if(id_existing_FoF_group == id_FoF_group) continue;
765 
766  // We merge the current FoF group with an existing one
767  size_t new_FoF_id = std::min(id_FoF_group, id_existing_FoF_group);
768 #ifdef USE_OMP
769 #pragma omp parallel for
770 #endif
771  for(size_t jj = 0; jj < NumPart; jj++)
772  if(particle_id_FoF[jj] == id_FoF_group || particle_id_FoF[jj] == id_existing_FoF_group)
773  particle_id_FoF[jj] = new_FoF_id;
774 
775  // The current group is set to be the exisiting group
776  id_FoF_group = new_FoF_id;
777 
778  }
779  }
780  }
781  }
782  }
783 
784  //=============================================================================
785  // Change the FoF ID so its unique across tasks
786  //=============================================================================
787  size_t FoF_id_start_local = 0;
788 #ifdef USE_MPI
789  std::vector<long long int> max_FoF_id_over_tasks(FML::NTasks,0);
790  max_FoF_id_over_tasks[FML::ThisTask] = nFoF;
791  MPI_Allreduce(MPI_IN_PLACE, max_FoF_id_over_tasks.data(), FML::NTasks, MPI_LONG_LONG, MPI_SUM, MPI_COMM_WORLD);
792 
793  // Update FoF ID so its unique over tasks
794  for(int i = 1; i < FML::NTasks; i++){
795  max_FoF_id_over_tasks[i] += max_FoF_id_over_tasks[i-1];
796  }
797  if(FML::ThisTask > 0)
798  FoF_id_start_local = size_t(max_FoF_id_over_tasks[FML::ThisTask-1]);
799 #endif
800 
801 #ifdef USE_OMP
802 #pragma omp parallel for
803 #endif
804  for(size_t jj = 0; jj < NumPart; jj++)
805  if(particle_id_FoF[jj] != no_FoF_ID)
806  particle_id_FoF[jj] += FoF_id_start_local;
807 
808 #ifdef DEBUG_FOF
809  std::cout << "FoFIndex on " << FML::ThisTask << " is in the range " << FoF_id_start_local << " -> " << FoF_id_start_local + nFoF-1 << "\n";
810 #endif
811 
812  }

814  template<class T, int NDIM, class FoFHaloClass = FoFHalo<T,NDIM>>
815  void FriendsOfFriends(
816  T *part,
817  size_t NumPart,
818  double fof_distance,
819  int nmin_FoF_group,
820  bool periodic,
821  std::vector<FoFHaloClass> &LocalFoFGroups){
822 
823  //============================================================================
824  // If we can assume a FoF group do not span more than 2 tasks then
825  // we can merge in parallel which is faster, but will be wrong if
826  // the FoF groups are too wide
827  //============================================================================
828 
829  const bool merging_in_parallel = merging_in_parallel_default;
830 
831  //============================================================================
832  // Make a grid where the grid-size is atleast fof_distance to ease the linking
833  // For convenience with parallelization adjust so the grid is divided by ntasks
834  //============================================================================
835 
836  int Ngrid = int(1.0 / fof_distance);
837  Ngrid = Ngrid - Ngrid % FML::NTasks;
838  if(Ngrid % FML::NTasks > Ngrid/2) Ngrid *= 2;
839  if(Ngrid > FoF_Ngrid_max) Ngrid = (FoF_Ngrid_max/FML::NTasks) * FML::NTasks;
840  if(Ngrid < 3 * FML::NTasks) Ngrid = 3 * FML::NTasks;
841  assert(1.0/Ngrid > fof_distance);
842  const int Local_nx = Ngrid / FML::NTasks;
843 
844 #ifdef DEBUG_FOF
845  if(FML::ThisTask == 0){
846  std::cout << "FriendsOfFriends linking\n";
847  std::cout << "FoF Linking Distance: " << fof_distance << "\n";
848  std::cout << "FoF Gridsize Ngrid = " << Ngrid << " Local: " << Local_nx << "\n";
849  }
850 #endif
851 
852  //=============================================================================
853  // Add particles to cells to speed up the linking below
854  //=============================================================================
855  std::vector<FoFCells> PartCells;
856  ParticlesToCells<T,NDIM>(
857  Ngrid,
858  part,
859  NumPart,
860  PartCells);
861 
862  //=============================================================================
863  // Do local FoF linking
864  //=============================================================================
865  std::vector<size_t> particle_id_FoF;
866  FriendsOfFriendsLinkingLocal<T,NDIM>(
867  part,
868  NumPart,
869  PartCells,
870  Ngrid,
871  Local_nx,
872  fof_distance,
873  particle_id_FoF,
874  periodic);
875 
876  //=============================================================================
877  // Link across boundaries
878  // Index in part of the particles that have a link across the boundary on the right
879  // and the index of the FoF group these belong to
880  // isShared is the index of particles that has a link to boundary particles on the left
881  //=============================================================================
882 
883  std::vector<size_t> BoundaryParticleIndex;
884  std::vector<size_t> BoundaryParticleRightFoFIndex;
885  std::vector<char> isShared(NumPart,0);
886  BoundaryLinking<T,NDIM>(
887  fof_distance,
888  part,
889  NumPart,
890  Ngrid,
891  PartCells,
892  particle_id_FoF,
893  BoundaryParticleIndex,
894  BoundaryParticleRightFoFIndex,
895  isShared,
896  periodic,
897  merging_in_parallel);
898 
899  // Free memory no longer needed
900  PartCells.clear();
901  PartCells.shrink_to_fit();
902 
903  // Deal with the left boundary. First figure out FoFIDs of boundary particles
904  std::vector<size_t> BoundaryParticleLeftFoFIndex;
905  for(size_t i = 0; i < NumPart; i++){
906  if(isShared[i] == 1){
907  BoundaryParticleLeftFoFIndex.push_back(particle_id_FoF[i]);
908  }
909  }
910 
911  // ...and make sure we only have unique FoFIDs
912  std::vector<size_t>::iterator it;
913  it = std::unique(BoundaryParticleLeftFoFIndex.begin(), BoundaryParticleLeftFoFIndex.end());
914  BoundaryParticleLeftFoFIndex.resize(std::distance(BoundaryParticleLeftFoFIndex.begin(),it));
915 
916  //==========================================================
917  // Count number of FoF groups on local task and the number
918  // of particles in the group. Also record which FoF groups
919  // that are shared across tasks
920  //==========================================================
921 
922  // Sort the FoFIDs from small to large and count how many local FoF groups we have
923  std::vector<size_t> tmp;
924  for(auto &i : particle_id_FoF)
925  if(i != no_FoF_ID) tmp.push_back(i);
926  std::sort(tmp.begin(), tmp.end(), [](const size_t & a, const size_t & b) -> bool{ return a < b;});
927 
928  size_t curFoFID = tmp[0];
929  size_t count = 0;
930  size_t groupsize = 1;
931  std::vector<size_t> FoFIndex;
932  std::vector<size_t> ningroup;
933  for(size_t i = 1; i < NumPart; i++){
934  if(tmp[i] == no_FoF_ID) continue;
935  if(tmp[i] == curFoFID and i != NumPart-1){
936  groupsize++;
937  } else {
938  ningroup.push_back(groupsize);
939  FoFIndex.push_back(curFoFID);
940  curFoFID = tmp[i];
941  groupsize = 1;
942  count++;
943  }
944  }
945  tmp.clear();
946  tmp.shrink_to_fit();
947 
948  // Compute what are the shared groups
949  std::vector<char> GroupIsShared(ningroup.size(), 0);
950  size_t nshared_groups = 0;
951  for(size_t i = 0; i < FoFIndex.size(); i++){
952  for(size_t j = 0; j < BoundaryParticleRightFoFIndex.size(); j++){
953  if( BoundaryParticleRightFoFIndex[j] == FoFIndex[i] ){
954  GroupIsShared[i] = 1;
955  nshared_groups++;
956  }
957  }
958  for(size_t j = 0; j < BoundaryParticleLeftFoFIndex.size(); j++){
959  if( BoundaryParticleLeftFoFIndex[j] == FoFIndex[i] ){
960  GroupIsShared[i] = 1;
961  nshared_groups++;
962  }
963  }
964  }
965  BoundaryParticleLeftFoFIndex.clear();
966  BoundaryParticleRightFoFIndex.clear();
967  BoundaryParticleLeftFoFIndex.shrink_to_fit();
968  BoundaryParticleRightFoFIndex.shrink_to_fit();
969 
970  // Count local non-shared groups
971  int nnonshared = 0;
972  std::vector<size_t> LocalNonSharedHalos;
973  for(size_t i = 0; i < ningroup.size(); i++){
974  if(ningroup[i] >= size_t(nmin_FoF_group) and GroupIsShared[i] == 0){
975  nnonshared++;
976  LocalNonSharedHalos.push_back(FoFIndex[i]);
977  }
978  }
979 
980  // Sum up total number of local non-shared groups
981  int non_shared_total = nnonshared;
982 
983 #ifdef USE_MPI
984  MPI_Allreduce(MPI_IN_PLACE, &non_shared_total, 1, MPI_INT, MPI_SUM, MPI_COMM_WORLD);
985 #endif
986 #ifdef DEBUG_FOF
987  std::cout << FML::ThisTask << " has " << non_shared_total << " nonshared halos\n";
988  std::cout << FML::ThisTask << " has " << count << " FoF groups before merging. The number of shared groups " << nshared_groups << " Nlocal groups: " << "\n";
989  std::cout << FML::ThisTask << " has " << BoundaryParticleRightFoFIndex.size() << " boundary particles\n";
990 #endif
991 
992 #ifdef USE_MPI
993  // We are here is we have more than 1 task
994 
995  // Gather data to send
996  std::vector<size_t> FoFIDSharedGroup;
997  std::vector<size_t> ninSharedGroup;
998  for(size_t i = 0; i < GroupIsShared.size(); i++){
999  if(GroupIsShared[i] == 1){
1000  FoFIDSharedGroup.push_back(FoFIndex[i]);
1001  ninSharedGroup.push_back(ningroup[i]);
1002  }
1003  }
1004  GroupIsShared.clear();
1005  GroupIsShared.shrink_to_fit();
1006 
1007  // Gather data about shared groups
1008  std::vector<int> shared_groups_in_task(FML::NTasks, 0);
1009  shared_groups_in_task[FML::ThisTask] = ninSharedGroup.size();
1010  MPI_Allreduce(MPI_IN_PLACE, shared_groups_in_task.data(), FML::NTasks, MPI_INT, MPI_SUM, MPI_COMM_WORLD);
1011  std::vector< std::vector<size_t> > FoFIDSharedGroupFromOtherTasks(FML::NTasks);
1012  std::vector< std::vector<size_t> > ninSharedGroupFromOtherTasks(FML::NTasks);
1013  if(FML::ThisTask == 0){
1014  FoFIDSharedGroupFromOtherTasks[0] = FoFIDSharedGroup;
1015  ninSharedGroupFromOtherTasks[0] = ninSharedGroup;
1016  }
1017 
1018  // Send from task i and recieve in task 0, the rest just sit quiet and wait
1019  for(int i = 1; i < FML::NTasks; i++){
1020  const int bytes = sizeof(size_t) * shared_groups_in_task[i];
1021  MPI_Barrier(MPI_COMM_WORLD);
1022  if(FML::ThisTask == i){
1023  MPI_Send(FoFIDSharedGroup.data(), bytes, MPI_BYTE, 0, 0, MPI_COMM_WORLD);
1024  MPI_Send(ninSharedGroup.data(), bytes, MPI_BYTE, 0, 0, MPI_COMM_WORLD);
1025  }
1026  if(FML::ThisTask == 0){
1027  FoFIDSharedGroupFromOtherTasks[i] = std::vector<size_t>(shared_groups_in_task[i]);
1028  ninSharedGroupFromOtherTasks[i] = std::vector<size_t>(shared_groups_in_task[i]);
1029  MPI_Status status;
1030  MPI_Recv(FoFIDSharedGroupFromOtherTasks[i].data(), bytes, MPI_BYTE, i, 0, MPI_COMM_WORLD, &status);
1031  MPI_Recv(ninSharedGroupFromOtherTasks[i].data(), bytes, MPI_BYTE, i, 0, MPI_COMM_WORLD, &status);
1032  }
1033  }
1034 
1035  // Count total number of halos
1036  std::vector<size_t> SharedHalos;
1037  std::vector<size_t> nSharedHalos;
1038 
1039  // This variable only correct on task 0
1040  int ntotal_halos = 0;
1041  if(FML::ThisTask == 0){
1042  int nshared_halos = 0;
1043 
1044  // Make a map to count up the number of particles in each shared group
1045  std::map<size_t, size_t> groupmap;
1046  for(int i = 0; i < FML::NTasks; i++){
1047  for(int j = 0; j < shared_groups_in_task[i]; j++){
1048  size_t fofid = FoFIDSharedGroupFromOtherTasks[i][j];
1049  size_t n = ninSharedGroupFromOtherTasks[i][j];
1050  groupmap[fofid] += n;
1051  }
1052  }
1053 
1054  // Go through the map and check if any is larger.
1055  for(auto &it : groupmap){
1056  if(it.second >= size_t(nmin_FoF_group)){
1057 #ifdef DEBUG_FOF
1058  std::cout << "Shared group " << it.first << " has " << it.second << " particles\n";
1059 #endif
1060  SharedHalos.push_back(it.first);
1061  nSharedHalos.push_back(it.second);
1062  nshared_halos++;
1063  }
1064  }
1065 #ifdef DEBUG_FOF
1066  std::cout << "We have " << nshared_halos << " acceptable shared halos for a total of " << nshared_halos + non_shared_total << "\n";
1067 #endif
1068 
1069  ntotal_halos = nshared_halos + non_shared_total;
1070  }
1071  FoFIDSharedGroupFromOtherTasks.clear();
1072  FoFIDSharedGroupFromOtherTasks.shrink_to_fit();
1073 
1074  // Communicate number of groups to all tasks
1075  int nsharedhalos = SharedHalos.size();
1076  MPI_Bcast( &nsharedhalos, 1, MPI_INT, 0, MPI_COMM_WORLD);
1077  SharedHalos.resize(nsharedhalos);
1078  MPI_Bcast(SharedHalos.data(), sizeof(size_t) * nsharedhalos, MPI_BYTE, 0, MPI_COMM_WORLD);
1079 
1080  // Figure out which shared halos we have a part of on current task
1081  std::vector<size_t> LocalSharedHalos;
1082  for(int i = 0; i < nsharedhalos; i++){
1083  for(auto &id : FoFIDSharedGroup)
1084  if(id == SharedHalos[i]){
1085  LocalSharedHalos.push_back(id);
1086  }
1087  }
1088  SharedHalos.clear();
1089  SharedHalos.shrink_to_fit();
1090 #else
1091  // No MPI no shared halos
1092  std::vector<size_t> LocalSharedHalos;
1093  size_t ntotal_halos = non_shared_total;
1094 #endif
1095 
1096  //=============================================================================
1097  // Create the halos (here we will have redunancies as we havent merged shared halos yet)
1098  //=============================================================================
1099 
1100  std::vector<FoFHaloClass> halos;
1101  halos.reserve(LocalSharedHalos.size() + LocalNonSharedHalos.size());
1102 
1103  // Process non-shared halos
1104  for(auto &id : LocalNonSharedHalos){
1105  const bool shared = false;
1106  FoFHaloClass g(id, shared);
1107  for(size_t i = 0; i < NumPart; i++){
1108  if(particle_id_FoF[i] == id){
1109  g.add(part[i], periodic);
1110  }
1111  }
1112  halos.push_back(g);
1113  }
1114 
1115  // Process shared halos
1116  for(auto &id : LocalSharedHalos){
1117  const bool shared = true;
1118  FoFHaloClass g(id, shared);
1119  for(size_t i = 0; i < NumPart; i++){
1120  if(particle_id_FoF[i] == id){
1121  g.add(part[i], periodic);
1122  }
1123  }
1124  halos.push_back(g);
1125  }
1126  LocalSharedHalos.clear();
1127  LocalNonSharedHalos.clear();
1128  LocalSharedHalos.shrink_to_fit();
1129  LocalNonSharedHalos.shrink_to_fit();
1130 
1131  // Send halos to task 0 which merges and returns
1132  std::vector<FoFHaloClass> myhalos = halos;
1133 #ifdef USE_MPI
1134  std::vector<int> nhalosontask(FML::NTasks, 0);
