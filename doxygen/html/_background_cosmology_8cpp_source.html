17 
18  const double rhoc0_over_h2 = 3.0 * Constants.H0_over_h * Constants.H0_over_h / (8.0 * M_PI * Constants.G);

22  double OmegaNuh2 = (7.0 / 8.0) * Neff * pow(Tnu_over_TCMB, 4) * OmegaRh2;
23  double OmegaBh2, OmegaCDMh2, OmegaKh2, OmegaLambdah2, h2;
24 
25  if(PhysicalParameters){
26  OmegaBh2 = p.get<double>("OmegaBh2");
27  OmegaCDMh2 = p.get<double>("OmegaCDMh2");
28  OmegaKh2 = p.get<double>("OmegaKh2");
29  OmegaLambdah2 = p.get<double>("OmegaLambdah2");
30  h2 = OmegaBh2 + OmegaCDMh2 + OmegaKh2 + OmegaRh2 + OmegaNuh2 + OmegaLambdah2;
31  assert(h2 > 0.0);
32  } else {
33  h2 = pow(p.get<double>("h"), 2);
34  OmegaBh2 = p.get<double>("OmegaB") * h2;
35  OmegaCDMh2 = p.get<double>("OmegaCDM") * h2;
36  OmegaKh2 = p.get<double>("OmegaK") * h2;
37  OmegaLambdah2 = (h2 - OmegaBh2 - OmegaCDMh2 - OmegaKh2 - OmegaRh2 - OmegaNuh2);
38  }
39 
40  // If OmegaKh2 is too negative then H = 0 at some point we need OmegaK > -(OmegaLambda OmegaM^2)^1/3
41  const double min_OmegaKh2 = -pow(OmegaLambdah2 * 27/4. * (OmegaCDMh2+OmegaBh2) * (OmegaCDMh2+OmegaBh2),0.33333);
42  if(OmegaKh2 < min_OmegaKh2) {
43  throw std::runtime_error("OmegaKh2 = " + std::to_string(OmegaKh2) + " is too negative (must be > "
44  + std::to_string(min_OmegaKh2) + " ). The Hubble factor will be zero at some point\n");
45  }
46 
47  // Tiny unobservable values of the curvature is simply set to zero
48  if(fabs(OmegaKh2/h2) < 1e-6){
49  OmegaLambdah2 += OmegaKh2;
50  OmegaKh2 = 0.0;
51  }
52 
53  // Tiny unobservable values of dark energy is simply set to zero
54  if(fabs(OmegaLambdah2/h2) < 1e-6){
55  OmegaCDMh2 += OmegaLambdah2;
56  OmegaLambdah2 = 0.0;
57  }
58

60  OmegaCDM = OmegaCDMh2 / (h2);
61  OmegaK = OmegaKh2 / (h2);
62  OmegaLambda = OmegaLambdah2 / (h2);
63  OmegaR = OmegaRh2 / (h2);
64  OmegaNu = OmegaNuh2 / (h2);
65  OmegaRtot = OmegaR + OmegaNu;
66  OmegaM = OmegaB + OmegaCDM;
67  h = sqrt(h2);
68  H0 = h * Constants.H0_over_h;
69  K = -OmegaK * H0 * H0 / (Constants.c * Constants.c);
70  }

134  void BackgroundCosmology::compute_background(){
135 
136  // Make a array of log(a) from the very early Univers till a bit into the
137  // future
138  DVector x_array = FML::MATH::linspace(x_min_background, x_max_background, n_pts_splines);
139 
140  // Define the Hubble functions we are to spline
141  auto H_function = [&](double x) {
142  return H0 * sqrt(OmegaLambda + OmegaK * exp(-2 * x) + OmegaM * exp(-3 * x) + OmegaRtot * exp(-4 * x));
143  };
144  auto Hp_function = [&](double x) {
145  return exp(x) * H_function(x);
146  };
147  auto dHdx_function = [&](double x) {
148  return 1.0 / (2.0 * H_function(x)) * H0 * H0 * (-2 * OmegaK * exp(-2 * x) - 3 * OmegaM * exp(-3 * x) - 4 * OmegaRtot * exp(-4 * x));
149  };
150  auto dHpdx_function = [&](double x) {
151  return 1.0 / (2.0 * Hp_function(x)) * H0 * H0 * (2 * OmegaLambda * exp(2 * x) - OmegaM * exp(-x) - 2 * OmegaRtot * exp(-2 * x));
152  };
153  auto w_function = [&](double x) {
154  return (OmegaRtot * exp(-4 * x)/3.0 - OmegaLambda -OmegaK * exp(-2 * x)/3.0) / pow(H_function(x)/H0,2);
155  };
156 
157  // Spline the Hubble function and derivatives
158  DVector H_array(x_array);
159  DVector Hp_array(x_array);
160  DVector dHdx_array(x_array);
161  DVector dHpdx_array(x_array);
162  DVector w_array(x_array);
163  for (auto& x : H_array) {
164  double xs = x;
165  x = H_function(x);
166  if(x != x){
167  throw std::runtime_error("Compute background. H(x) crossing 0.0 at x = " + std::to_string(xs) + "\n");
168  }
169  }
170  for (auto& x : Hp_array) {
171  x = Hp_function(x);
172  }
173  for (auto& x : dHdx_array) {
174  x = dHdx_function(x);
175  }
176  for (auto& x : dHpdx_array) {
177  x = dHpdx_function(x);
178  }
179  for (auto& x : w_array) {
180  x = w_function(x);
181  }
182  H_spline.create(x_array, H_array, "H_of_x");
183  Hp_spline.create(x_array, Hp_array, "Hp_of_x");
184  dHdx_spline.create(x_array, dHdx_array, "dHdx_of_x");
185  dHpdx_spline.create(x_array, dHpdx_array, "dHpdx_of_x");
186  w_spline.create(x_array, w_array, "w_of_x");
187  }

189  void BackgroundCosmology::compute_conformal_time(){
190 
191  // Make a array of log(a) from the very early Univers till a bit into the
192  // future
193  DVector x_array = FML::MATH::linspace(x_min_background, x_max_background, n_pts_splines);
194 
195  // The ODE system deta/dx = c/Hp and dt/dx = c/H
196  ODEFunction deriv = [&](double x, const double* y, double* dydx) {
197  double HpoverH0 = Hp_of_x(x) / H0;
198 
199  // Conformal time deta/dx in units of c/H0
200  dydx[0] = 1.0 / HpoverH0;
201 
202  // dt/dx (Age of the Universe) in units of 1/H0
203  dydx[1] = exp(x) / HpoverH0;
204  return GSL_SUCCESS;
205  };
206 
207  // The initial conditions
208  DVector eta_initial{0.0, 0.0};
209  if( OmegaRtot > 0.0 )
210  eta_initial = { exp(x_array[0]) / sqrt(OmegaRtot),
211  exp(2.0 * x_array[0]) / 2.0 / sqrt(OmegaRtot) };
212 
213  // Solve the ODE
214  ODESolver eta_ode(FIDUCIAL_COSMO_HSTART_ODE_ETA,
215  FIDUCIAL_COSMO_HSTART_ODE_ETA,
216  FIDUCIAL_COSMO_RELERR_ODE_ETA);
217  eta_ode.solve(deriv, x_array, eta_initial);
218 
219  // Fetch the result and get eta and t. Divide by H0 as we solved in units of
220  // 1/H0
221  auto eta = eta_ode.get_data_by_component(0);
222  for (auto& y : eta) {
223  y *= Constants.c / H0;
224  }
225  auto age = eta_ode.get_data_by_component(1);
226  for (auto& y : age) {
227  y *= 1.0 / H0;
228  }
229 
230  // Spline up eta and the age of the Universe
231  eta_of_x_spline.create(x_array, eta, "eta_of_x_spline");
232  cosmic_time_of_x_spline.create(x_array, age, "cosmic_time_of_x_spline");
233 
234  // Remember to set eta0
235  eta0 = eta_of_x_spline(0.0);
236 
237  // Comoving distance (only to the present time and not too far back)
238  DVector chi_array(eta);
239  DVector x_array_chi = FML::MATH::linspace(std::max(x_min_background,-15.0), 0.0, n_pts_splines);
240  for(int i = 0; i < n_pts_splines; i++) {
241  chi_array[i] = (eta0 - eta_of_x_spline(x_array_chi[i]));
242  }
243  chi_of_x_spline.create(x_array_chi, chi_array);
244  x_of_chi_spline.create(chi_array, x_array_chi);
245  }

247  void BackgroundCosmology::compute_growth_factors(){
248  // This is the growth factor for DeltaM, the total comoving matter perturbationvwhich includes radiation
249  // If the flag below is true its the usual one
250  const bool cdm_growth_fac = true;
251 
252  DVector x_array;
253  if(cdm_growth_fac)
254  x_array = FML::MATH::linspace(-log(1000.0), x_max_background, n_pts_splines);
255  else
256  x_array = FML::MATH::linspace(x_min_background, x_max_background, n_pts_splines);
257 
258  // Growth factors (1LPT and 2LPT)
259  ODEFunction deriv_growth = [&](double x, const double* y, double* dydx) {
260  const double D1 = y[0];
261  const double V1 = y[1];
262  const double D2 = y[2];
263  const double V2 = y[3];
264  const double H = H_of_x(x) / H0;
265  const double a = exp(x), a2 = a*a;
266  const double Omega = get_OmegaM(x) + (cdm_growth_fac ? 0.0 : get_OmegaRtot(x));
267  dydx[0] = V1 / (H * a2);
268  dydx[1] = 1.5 * Omega * (H * a2) * D1;
269  dydx[2] = V2 / (H * a2);
270  dydx[3] = 1.5 * Omega * (H * a2) * (D2 - D1*D1);
271  return GSL_SUCCESS;
272  };
273 
274  // Initial conditions based on analytical solutions
275  // D1 = a^n, D2 = prefac * D1^2
276  const double xeq = log(OmegaRtot/OmegaM);
277  const double xini = x_array[0];
278  const double nindex = xini < xeq ? sqrt(1.5) : 1.0;
279  const double prefac = xini < xeq ? -1.0/3.0 : -3.0/7.0;
280  const double D1ini = 1.0;
281  const double dD1dxini = nindex * D1ini;
282  const double D2ini = prefac * D1ini * D1ini;
283  const double dD2dxini = 2.0 * prefac * D1ini * dD1dxini;
284  const double V1ini = dD1dxini * (H_of_x(xini) / H0 * exp(2.0*xini));
285  const double V2ini = dD2dxini * (H_of_x(xini) / H0 * exp(2.0*xini));
286 
287  // Solve the ODE
288  DVector D_ini{D1ini, V1ini, D2ini, V2ini};
289  ODESolver growth_ode(1e-3, 1e-12, 1e-12);
290  growth_ode.solve(deriv_growth, x_array, D_ini);
291 
292  // Fetch solution
293  auto D1 = growth_ode.get_data_by_component(0);
294  auto V1 = growth_ode.get_data_by_component(1);
295  auto D2 = growth_ode.get_data_by_component(2);
296  auto V2 = growth_ode.get_data_by_component(3);
297 
298  // Normalize to unity at present time and take the log
299  DVector dD1dx(x_array.size());
300  DVector dD2dx(x_array.size());
301  for(size_t i = 0; i < x_array.size(); i++){
302  D1[i] /= D1[x_array.size()-1];
303  V1[i] /= D1[x_array.size()-1];
304  D2[i] /= D2[x_array.size()-1];
305  V2[i] /= D2[x_array.size()-1];
306  dD1dx[i] = V1[i] / (H_of_x(x_array[i]) / H_of_x(0.0) * exp(2.0*x_array[i]));
307  dD2dx[i] = V2[i] / (H_of_x(x_array[i]) / H_of_x(0.0) * exp(2.0*x_array[i]));
308  }
309 
310  // Spline up
311  D1_spline.create(x_array, D1, "D1_spline");
312  D2_spline.create(x_array, D2, "D2_spline");
313  }
