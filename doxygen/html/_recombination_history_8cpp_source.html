87  const double m_e = Constants.m_e;
88  const double hbar = Constants.hbar;
89  const double m_H = Constants.m_H;

92

94  const double OmegaB = cosmo->get_OmegaB();
95  const double h = cosmo->get_h();
96  const double TCMB_of_x = cosmo->get_TCMB(x);
97  const double H0 = h * H0_over_h;
98

100  const double kT_b = k_b * TCMB_of_x;

102

105

107  double f_e = 1.0;

109  f_e = 1.0 - 1.0 / saha_factor;
110  } else {

112  }
113

115  if(a < 1e-10) f_e = 1.0;
116 
117  // Return electron fraction and number density
118  const double Xe = f_e;
119  const double ne = f_e * n_b;
120 
121  return std::pair<double, double>(Xe, ne);

130

132  const double k_b = Constants.k_b;
133  const double G = Constants.G;
134  const double m_e = Constants.m_e;
135  const double hbar = Constants.hbar;
136  const double m_H = Constants.m_H;

138  const double H0_over_h = Constants.H0_over_h;
139  const double xhi0 = Constants.xhi0;
140  const double xhi1 = Constants.xhi1;
141 
142  // Cosmological parameters
143  const double OmegaB = cosmo->get_OmegaB();
144  const double h = cosmo->get_h();
145  const double TCMB_of_x = cosmo->get_TCMB(x);
146  const double H0 = h * H0_over_h;
147 
148  // To make the code run with OmegaB = 0
149  if (OmegaB == 0.0) {
150  return std::pair<double, double>(1.0, 1e-100);
151  }
152 
153  // Baryon temperatur and number density
154  const double kT_b = k_b * TCMB_of_x;
155  const double n_b = (OmegaB * 3.0 * H0 * H0) / (8 * M_PI * G * m_H * a * a * a);
156 
157  // Prefactor for rhs of Saha equations
158  const double saha_factor = pow(m_e * kT_b / (hbar * hbar * 2.0 * M_PI), 1.5) / n_b;
159 
160  // No point of solving if the result is basically 0
161  double factor = sqrt(saha_factor * exp(-epsilon_0 / kT_b) / (1.0 - Yp));
162  if (factor < 1e-5)
163  return { factor, (1.0 - Yp) * factor * n_b };
164 
165  // Iterative method for finding x_He_plus, x_He_plusplus, x_H_plus
166  double f_e = 1.0;
167  double f_e_old = 0.0;
168  while (abs(f_e - f_e_old) > 1e-10) {
169  // R.h.s of the Saha equations for He++, He+ and H+ respectivily
170  const double rhs_x_He_plus = 2.0 * saha_factor * exp(-xhi0 / kT_b) / f_e ;
171  const double rhs_x_He_plusplus = 4.0 * saha_factor * exp(-xhi1 / kT_b) / f_e;
172  const double rhs_x_H_plus = saha_factor * exp(-epsilon_0 / kT_b) / f_e;
173 
174  // Abundances of He++, He+ and H+ respectivily
175  const double x_He_plus = rhs_x_He_plus / (1.0 + rhs_x_He_plus + rhs_x_He_plus * rhs_x_He_plusplus);
176  const double x_He_plusplus = rhs_x_He_plusplus * x_He_plus;
177  const double x_H_plus = rhs_x_H_plus / (1.0 + rhs_x_H_plus);
178 
179  // Calculate new value of f_e
180  f_e_old = f_e;
181  f_e = (2.0 * x_He_plusplus + x_He_plus) * Yp / 4.0 + (1.0 - Yp) * x_H_plus;
182  }
183 
184  // Just in case we want to compute this with an extremely early starting x
185  if(a < 1e-10) f_e = 1.0;
186 
187  // Return electron fraction and number density
188  const double Xe = f_e / (1.0 - Yp);
189  const double ne = f_e * n_b;
190 
191  return { Xe, ne };

260 
261  // Settings for the arrays we use below
262  const int npts = npts_Xe_array;
263  const double x_start = x_start_rec_array;

269  DVector ne_array(npts);
270  DVector Xe_saha_arr(npts);
271  DVector Tb_array(npts);
272  DVector cs2_baryon_array(npts);
273 
274  // Saha calculation
275  for (int i = 0; i < npts; i++) {
276  //==============================================================
277  // Electron fraction and number density from Saha equation
278  //==============================================================
279  auto Xe_ne_data = electron_fraction_from_saha_equation_with_helium(x_array[i]);
280  Xe_saha_arr[i] = Xe_ne_data.first;
281  ne_array[i] = Xe_ne_data.second;
282  Tb_array[i] = cosmo->get_TCMB(x_array[i]);
283  }
284 
285  // Calculate recombination history
286  for (int i = 0; i < npts; i++) {
287  auto Xe_current = Xe_saha_arr[i];
288 
289  // Two regimes: Saha and Peebles regime
290  if (Xe_saha_arr[i] >= Xe_saha_limit) {
291  Xe_array[i] = Xe_current;
292  } else {
293 
294  //==============================================================
295  // We need to solve the Peebles equation for the rest of the time
296  //==============================================================
297 
298  // Constants and physical parameters needed below
299  const double m_H = Constants.m_H;
300  const double G = Constants.G;
301  const double OmegaB = cosmo->get_OmegaB();
302  const double H0 = cosmo->get_H0();
303  const double n_b0 = (OmegaB * 3.0 * H0 * H0) / (8 * M_PI * G * m_H);
304 
305  // Make x-array for Peebles system from current time till the end
306  const auto first = x_array.begin() + i;
307  const auto last = x_array.end();
308  DVector x_array_peebles(first, last);
309 
310  // The Peebles ODE equation
311  ODEFunction deriv = [&](double x, const double* y, double* dydx) {
312  return rhs_peebles_ode(x, y, dydx);
313  };
314 
315  // Set initial conditions for { X_e, (T_b/T_gamma-1) }
316  DVector Xe_initial { Xe_current, 0.0 };
317 
318  // Solve the Peebles ODE
319  ODESolver peebles_Xe_ode(FIDUCIAL_HSTART_ODE_PEEBLES,
320  FIDUCIAL_ABSERR_ODE_PEEBLES,
321  FIDUCIAL_RELERR_ODE_PEEBLES);
322  peebles_Xe_ode.solve(deriv, x_array_peebles, Xe_initial);
323 
324  // Fetch the Xe solution
325  const auto data = peebles_Xe_ode.get_data_by_component(0);
326 
327  // Fill up array with the result
328  for (int j = i; j < npts; j++) {
329  const double X_e = data[j - i];
330  Xe_array[j] = X_e;
331  ne_array[j] = (1.0 - Yp) * X_e * (n_b0 * exp(-3 * x_array[j]));
332  }
333 
334  // Fetch the computed baryon temperature
335  const auto baryon_temp = peebles_Xe_ode.get_data_by_component(1);
336  for (int j = 0; j < npts; j++) {
337  double Tb_of_x = cosmo->get_TCMB(x_array[j]);
338  if (j >= i) {
339  Tb_of_x *= (1.0 + baryon_temp[j - i]);
340  }
341  Tb_array[j] = Tb_of_x;
342  }
343 
344  // We are done so exit for loop
345  break;
346  }
347  }
348 
349 #ifdef USE_RECFAST
350  // Run RECFAST to get Xe and Tb and overwrite the resulrs we already have
351  if(userecfast) {
352  auto recfastdata = run_recfast(x_array);
353  Xe_array = recfastdata.first;
354  Tb_array = recfastdata.second;
355  }
356 #endif
357 
358  // Spline baryon temperature
359  Tb_spline.create(x_array, Tb_array, "Temp_baryon_of_x");
360 
361  // Make the baryon sound speed (cs/c)^2
362  // The mean moleculary weight - we just use m_H here for simplicity
363  // Add in the effects of reionization?
364  for (int j = 0; j < npts; j++) {

385  int RecombinationHistory::rhs_peebles_ode(double x,
386  const double* y,
387  double* dydx)
388  {
389  const double X_e = y[0];
390  const double a = exp(x);
391 
392  // Physical constants in SI units
393  const double k_b = Constants.k_b;
394  const double G = Constants.G;
395  const double c = Constants.c;
396  const double m_e = Constants.m_e;
397  const double hbar = Constants.hbar;
398  const double m_H = Constants.m_H;
399  const double sigma_T = Constants.sigma_T;
400  const double lambda_2s1s = Constants.lambda_2s1s;
401  const double epsilon_0 = Constants.epsilon_0;
402 
403  // Cosmological and recombination parameters
404  const double OmegaB = cosmo->get_OmegaB();
405  const double TCMB_of_x = cosmo->get_TCMB(x);
406  const double H0 = cosmo->get_H0();
407  const double H = cosmo->H_of_x(x);
408  const double Yp = get_Yp();
409 
410  // To be able to run the whole code without baryons
411  if (OmegaB == 0.0) {
412  dydx[0] = 0.0;
413  dydx[1] = 0.0;
414  return GSL_SUCCESS;
415  }
416 
417  // Baryon temperature and number density
418  const double n_b = (3.0 * OmegaB * H0 * H0) / (8 * M_PI * G * m_H * a * a * a);
419  const double n_H = (1.0 - Yp) * n_b;
420 
421  // Here we use the evolved baryon temperature. If we don't evolve this then
422  // an approx we can use is to set y = 0 below
423  const double kT_b = k_b * TCMB_of_x * (1.0 + y[1]);
424 
425  // Factors in the Peebles equation
426  const double eps_over_kT = epsilon_0 / kT_b;
427  const double exp_eps_over_kT = eps_over_kT < 200.0 ? exp(eps_over_kT) : exp(200.0);
428  const double phi_2 = 0.448 * log(eps_over_kT);
429  const double alpha_2 = phi_2 * (8.0 / sqrt(3.0 * M_PI)) * sigma_T * sqrt(eps_over_kT);
430  const double beta = alpha_2 * c * pow(m_e * kT_b / (2.0 * M_PI * hbar * hbar), 1.5) / exp_eps_over_kT;
431  const double beta_2 = beta * pow(exp_eps_over_kT, 0.75);
432  const double n1s = (1.0 - X_e) * n_H;
433  const double lambda_a = H * pow(3.0 * epsilon_0 / (hbar * c), 3) / (64.0 * M_PI * M_PI * n1s);
434  const double C_r = (lambda_2s1s + lambda_a) / (lambda_2s1s + lambda_a + beta_2);
435 
436  // Two different regimes to avoid numerical problems
437  dydx[0] = C_r / H * (beta * (1.0 - X_e) - c * n_H * alpha_2 * X_e * X_e);
438 
439  // Evolve the baryon temperature ODE (y is Tb/Tgamma-1 so 0 when tightly coupled)
440  const double R = 4.0 / 3.0 * cosmo->get_OmegaR() / OmegaB / a;
441  const double dtaudx = c * sigma_T / H * n_b * X_e;
442  dydx[1] = -y[1] - 1.0 - 2.0 * (Constants.m_H / Constants.m_e) * R * dtaudx * y[1];
443 
444  return GSL_SUCCESS;
445  }

498  // extra points in reionisation
499  DVector x_array(npts);
500  if (reionization) {
501  DVector x_array_before_reion = FML::MATH::linspace(x_start, x_start_reion, npts_before_reion);
502  DVector x_array_during_reion = FML::MATH::linspace(x_start_reion, x_end_reion, npts_during_reion);
503  DVector x_array_after_reion = FML::MATH::linspace(x_end_reion, x_end, npts_after_reion);
504 
505  // Combine to one array. Last point in previous array equals first point in
506  // new array so avoid duplications
507  double* arr1 = &x_array[0];
508  double* arr2 = &x_array[npts_before_reion - 1];
509  double* arr3 = &x_array[npts_before_reion - 1 + npts_during_reion - 1];
510  for (int i = 0; i < npts_before_reion; i++) {
511  arr1[i] = x_array_before_reion[i];
512  }
513  for (int i = 0; i < npts_during_reion; i++) {
514  arr2[i] = x_array_during_reion[i];
515  }
516  for (int i = 0; i < npts_after_reion; i++) {
517  arr3[i] = x_array_after_reion[i];
518  }
519 
520  // We integrate backwards from x=0 so reverse the array
521  std::reverse(x_array.begin(), x_array.end());
522  } else {
523  x_array = FML::MATH::linspace(x_end, x_start, npts);
524  }
525 
526  // The ODE system dtau/dx, dtau_noreion/dx and dtau_baryon/dx
527  ODEFunction deriv_tau = [&](double x, const double* y, double* dydx) {
528  const double OmegaB = cosmo->get_OmegaB();
529 
530  // If we don't have baryons then tau = 0
531  if (OmegaB == 0.0) {
532  dydx[0] = dydx[1] = dydx[2] = dydx[3] = dydx[4] = 0.0;
533  return GSL_SUCCESS;
534  }
535 
536  const double c = Constants.c;
537  const double sigma_T = Constants.sigma_T;
538  const double a = exp(x);
539  const double H = cosmo->H_of_x(x);
540  const double n_e = ne_of_x(x);
541  const double n_e_noreion = ne_of_x_noreion(x);
542  const double R = 4.0 / 3.0 * cosmo->get_OmegaR() / OmegaB / a;
543 
544  // Set the derivative for photon optical depth with and without reionization
545  dydx[0] = (-c * sigma_T * n_e / H);
546  dydx[1] = (-c * sigma_T * n_e_noreion / H);
547 
548  // The baryon optical depth
549  dydx[2] = -c * R * sigma_T * n_e_noreion / H;
550 
551  // Optical depth with Saha and with and without reionization
552  const double n_e_saha = ne_of_x_saha(x);
553  const double n_e_saha_noreion = ne_of_x_saha_noreion(x);
554  dydx[3] = (-c * sigma_T * n_e_saha / H);
555  dydx[4] = (-c * sigma_T * n_e_saha_noreion / H);
556 
557  return GSL_SUCCESS;
558  };
559 
560  // Set initial conditions (3 components: with and without reion + baryons)
561  DVector tau_initial(5, 0.0);
562 
563  // Solve the ODE
564  ODESolver tau_ode(FIDUCIAL_HSTART_ODE_TAU, FIDUCIAL_ABSERR_ODE_TAU,
565  FIDUCIAL_RELERR_ODE_TAU);
566  tau_ode.solve(deriv_tau, x_array, tau_initial, gsl_odeiv2_step_rk4);
567 
568  // Fetch the solution
569  auto tau_array = tau_ode.get_data_by_component(0);
570  auto tau_noreion_array = tau_ode.get_data_by_component(1);
571  auto tau_baryon_noreion_array = tau_ode.get_data_by_component(2);
572  auto tau_saha_array = tau_ode.get_data_by_component(3);
573  auto tau_saha_noreion_array = tau_ode.get_data_by_component(4);
574  auto dtaudx_array = tau_ode.get_derivative_data_by_component(0);
575  auto dtaudx_noreion_array = tau_ode.get_derivative_data_by_component(1);
576 
577  // Compute the visibility function
578  DVector g_tilde_array(npts);
579  DVector g_tilde_noreion_array(npts);
580  for (int i = 0; i < npts; i++) {
581  g_tilde_array[i] = -dtaudx_array[i] * exp(-tau_array[i]);
582  g_tilde_noreion_array[i] = -dtaudx_noreion_array[i] * exp(-tau_noreion_array[i]);
583  }
584 
585  // Make tau and gsplines
586  tau_of_x_spline.create(x_array, tau_array, "tau_of_x");
587  dtaudx_of_x_spline.create(x_array, dtaudx_array, "dtaudx_of_x");
588  g_tilde_of_x_spline.create(x_array, g_tilde_array, "g_tilde_of_x");
589  tau_of_x_noreion_spline.create(x_array, tau_noreion_array, "tau_of_x_noreion");
590  g_tilde_of_x_noreion_spline.create(x_array, g_tilde_noreion_array, "g_tilde_of_x_noreion");
591  tau_baryon_noreion_of_x_spline.create(x_array, tau_baryon_noreion_array, "tau_baryon_noreion_of_x");
592  tau_of_x_saha_spline.create(x_array, tau_saha_array, "tau_of_x_saha");
593  tau_of_x_saha_noreion_spline.create(x_array, tau_saha_noreion_array, "tau_of_x_saha_noreion");
594 
595  // Spline dgdx
596  DVector dgdx_tilde_array(g_tilde_array);
597  for (int i = 0; i < npts; i++) {
598  double x = x_array[i];
599  dgdx_tilde_array[i] = g_tilde_of_x_spline.deriv_x(x);
600  }
601  dgdx_tilde_of_x_spline.create(x_array, dgdx_tilde_array, "dgdx_tilde_of_x");
602 
603  // Spline ddgddx
604  DVector ddgddx_tilde_array(g_tilde_array);
605  for (int i = 0; i < npts; i++) {
606  double x = x_array[i];
607  ddgddx_tilde_array[i] = dgdx_tilde_of_x_spline.deriv_x(x);
608  }
609  ddgddx_tilde_of_x_spline.create(x_array, ddgddx_tilde_array,
610  "ddgddx_tilde_of_x");
611

614  void RecombinationHistory::solve_extra(){
615  const int npts = 1000;
616  const double x_start = x_start_rec_array;
617  const double x_end = x_end_rec_array;
618  DVector x_array = FML::MATH::linspace(x_start, x_end, npts);
619 
620  // The sound horizon ODE dr_s/dx = cs(x) / Hp(x)
621  ODEFunction deriv_sound = [&](double x, const double* y, double* dydx) {
622  if (cosmo->get_OmegaB() == 0.0) {
623  dydx[0] = 0.0;
624  } else {
625  dydx[0] = Constants.c * sqrt(get_sound_speed_squared(x)) / cosmo->Hp_of_x(x);
626  }
627  return GSL_SUCCESS;
628  };
629 
630  // The initial condition for the sound horizon ODE
631  DVector r_initial(1, 0.0);
632 
633  // Solve the sound horizon ODE
634  ODESolver r_ode(FIDUCIAL_HSTART_ODE_TAU, FIDUCIAL_ABSERR_ODE_TAU,
635  FIDUCIAL_RELERR_ODE_TAU);
636  r_ode.solve(deriv_sound, x_array, r_initial);
637 
638  // Fetch the data and make a spline
639  auto r_array = r_ode.get_data_by_component(0);
640  sound_horizon_of_x_spline.create(x_array, r_array, "sound_horizon_of_x");
641 
642  // Compute the diffusion scale
643  ODEFunction deriv_kd = [&](double x, const double* y, double* dydx) {
644  double a = exp(x);
645  double R = 4.0 / 3.0 * cosmo->get_OmegaR() / cosmo->get_OmegaB() / a;
646  double Hp = cosmo->Hp_of_x(x);
647  double rhs = -1.0 / 6.0 * (R * R + 16.0 * (1.0 + R) / 15.0) / ((1.0 + R) * (1.0 + R));
648  rhs /= dtaudx_of_x(x) * Hp * Hp / Constants.c / Constants.c;
649  dydx[0] = rhs;
650  return GSL_SUCCESS;
651  };
652  DVector kd_initial { 0.0 };
653  ODESolver kd_ode(FIDUCIAL_HSTART_ODE_TAU, FIDUCIAL_ABSERR_ODE_TAU,
654  FIDUCIAL_RELERR_ODE_TAU);
655  kd_ode.solve(deriv_kd, x_array, kd_initial);
656 
657  // Fetch the data and make a spline
658  auto kd_array = kd_ode.get_data_by_component(0);
659  for (auto& val : kd_array) {
660  if (val > 0.0)
661  val = 1.0 / sqrt(val);
662  }
663  kd_of_x_spline.create(x_array, kd_array, "damping scale");
664 
665  try {
666  // Find zstar (tau_noreion == 1)
667  x_star = FML::MATH::find_root_bisection(tau_of_x_noreion_spline, 1.0);
668 
669  // Find zstar (tau_noreion == 1)
670  x_star_saha = FML::MATH::find_root_bisection(tau_of_x_saha_noreion_spline, 1.0);
671 
672  // Find peak of visibility function
673  std::function<double(double)> func2 = [&](double x) {
674  return dtaudx_of_x(x) * dtaudx_of_x(x) - ddtauddx_of_x(x);
675  };
676  x_star2 = FML::MATH::find_root_bisection(func2, { -9.0, -6.0 });
677 
678  // Find zdrag (tau_baryon_noreion == 1)
679  x_drag = FML::MATH::find_root_bisection(tau_baryon_noreion_of_x_spline, 1.0);
680  } catch(...) {
681  std::cout << "Error computing decoupling time\n";
682  x_star = 0.0;
683  x_star_saha = 0.0;
684  x_star2 = 0.0;
685  x_star2 = 0.0;
686  }
687  }

785  // Implemention of reionization on Xe. This is the factor f in Xe = Xe^peebles + f
786  double RecombinationHistory::Xe_reionization_factor_of_x(double x) const
787  {
788  if (!reionization) {
789  return 0.0;
790  }
791 
792  const double y = exp(-1.5 * x);
793  const double yre = pow(1.0 + z_reion, 1.5);
794  const double delta_y_reion = 1.5 * sqrt(1.0 + z_reion) * delta_z_reion;
795  const double helium_fraction = 0.25 * Yp / (1.0 - Yp);
796 
797  // CAMB tanh reionization function
798  double f = (1.0 + helium_fraction) / 2.0 * (1.0 + tanh((yre - y) / delta_y_reion));
799 
800  // Take into account that Helium probably gets doubly reionized at low
801  // redshift
802  if (helium_reionization) {
803  const double z = exp(-x) - 1.0;
804  f += helium_fraction / 2.0 * (1.0 + tanh((z_helium_reion - z) / delta_z_helium_reion));
805  }
806  return f;
807  }
