515  template<class T>
516  void MPIParticles<T>::communicate_particles(){
517 #ifdef USE_MPI
518 
519  // The number of particles we start with
520  size_t NpartLocal_in_use_pre_comm = NpartLocal_in_use;
521 
522 #ifdef DEBUG_MPIPARTICLES
523  if(ThisTask == 0){
524  std::cout << "Communicating particles task: " << ThisTask << " Nparticles: " << NpartLocal_in_use_pre_comm << "\n" << std::flush;
525  }
526 #endif
527 
528  // Count how many particles to send to each task
529  // and move the particles to be send to the back of the array
530  // and reduce the NumPartLocal_in_use if a partice is to be sent
531  // After this is done we have all the particles to be send in
532  // location [NpartLocal_in_use, NpartLocal_in_use_pre_comm)
533  std::vector<int> n_to_send(NTasks,0);
534  std::vector<int> n_to_recv(NTasks,0);
535  size_t i = 0;
536  while(i < NpartLocal_in_use){
537  double x = p[i].get_pos()[0];
538  if(x >= x_max_per_task[ThisTask]){
539  int taskid = ThisTask;
540  while(1){
541  ++taskid;
542  if(x < x_max_per_task[taskid]) break;
543  }
544 
545  n_to_send[taskid]++;
546  swap_particles(p[i],p[--NpartLocal_in_use]);
547 
548  } else if(x < x_min_per_task[ThisTask]) {
549  int taskid = ThisTask;
550  while(1){
551  --taskid;
552  if(x >= x_min_per_task[taskid]) break;
553  }
554 
555  n_to_send[taskid]++;
556  swap_particles(p[i],p[--NpartLocal_in_use]);
557 
558  } else {
559  i++;
560  }
561  }
562 
563  // Communicate to get how many to recieve from each task
564  for(int i = 1; i < NTasks; i++){
565  int send_request_to = (ThisTask + i) % NTasks;
566  int get_request_from = (ThisTask - i + NTasks)% NTasks;
567 
568  // Send to the right, recieve from left
569  MPI_Status status;
570  MPI_Sendrecv(
571  &n_to_send[send_request_to], 1, MPI_INT, send_request_to, 0,
572  &n_to_recv[get_request_from], 1, MPI_INT, get_request_from, 0,
573  MPI_COMM_WORLD, &status);
574 
575  }
576 
577 #ifdef DEBUG_MPIPARTICLES
578  // Show some info
579  if(ThisTask == 0){
580  for(int i = 0; i < NTasks; i++){
581  std::cout << "Task " << ThisTask << " send to " << i << " : " << n_to_send[i] << "\n" << std::flush;
582  std::cout << "Task " << ThisTask << " recv from " << i << " : " << n_to_recv[i] << "\n" << std::flush;
583  }
584  }
585 #endif
586 
587  // Total number to send and recv
588  size_t ntot_to_send = 0;
589  size_t ntot_to_recv = 0;
590  for(int i = 0; i < NTasks; i++){
591  ntot_to_send += n_to_send[i];
592  ntot_to_recv += n_to_recv[i];
593  }
594

596  assert_mpi(NpartLocal_in_use_pre_comm == NpartLocal_in_use + ntot_to_send,

598 
599  // Allocate send buffer

601  std::vector<char> send_buffer(byte_per_particle * ntot_to_send);
602  std::vector<char> recv_buffer(byte_per_particle * ntot_to_recv);
603 
604  // Pointers to each send-recv place in the send-recv buffer
605  std::vector<size_t> offset_in_send_buffer(NTasks,0);
606  std::vector<size_t> offset_in_recv_buffer(NTasks,0);
607  std::vector<char*> send_buffer_by_task(NTasks,send_buffer.data());
608  std::vector<char*> recv_buffer_by_task(NTasks,recv_buffer.data());
609  for(int i = 1; i < NTasks; i++){

611  offset_in_recv_buffer[i] = offset_in_recv_buffer[i-1] + n_to_recv[i-1] * byte_per_particle;
612  send_buffer_by_task[i] = &send_buffer.data()[offset_in_send_buffer[i]];
613  recv_buffer_by_task[i] = &recv_buffer.data()[offset_in_recv_buffer[i]];
614  }
615 
616  // Gather particle data
617  for(size_t i = 0; i < ntot_to_send; i++){
618  size_t index = NpartLocal_in_use+i;
619  double x = p[index].get_pos()[0];
620  if(x >= x_max_per_task[ThisTask]){
621  int taskid = ThisTask;
622  while(1){
623  ++taskid;
624  if(x < x_max_per_task[taskid]) break;
625  }
626 
627  p[index].append_to_buffer(send_buffer_by_task[taskid]);
628  send_buffer_by_task[taskid] += byte_per_particle;
629 
630  } else if(x < x_min_per_task[ThisTask]) {
631  int taskid = ThisTask;
632  while(1){
633  --taskid;
634  if(x >= x_min_per_task[taskid]) break;
635  }
636 
637  p[index].append_to_buffer(send_buffer_by_task[taskid]);
638  send_buffer_by_task[taskid] += byte_per_particle;
639 
640  } else {
641 
642  // We should not be here as particles are moved
643  assert_mpi(false,
644  "[MPIParticles::communicate_particles] Error in communicate_particles. After moving particles we still have particles out of range\n");
645  }
646  }
647 
648  // We changed the send-recv pointers above so reset them
649  for(int i = 0; i < NTasks; i++){
650  send_buffer_by_task[i] = &send_buffer.data()[offset_in_send_buffer[i]];
651  recv_buffer_by_task[i] = &recv_buffer.data()[offset_in_recv_buffer[i]];
652  }
653 
654  // Communicate the particle data
655  for(int i = 1; i < NTasks; i++){
656  int send_request_to = (ThisTask + i) % NTasks;
657  int get_request_from = (ThisTask - i + NTasks)% NTasks;
658 
659  // Send to the right, recieve from left
660  MPI_Status status;
661  MPI_Sendrecv(
662  send_buffer_by_task[send_request_to],
663  n_to_send[send_request_to] * byte_per_particle,
664  MPI_CHAR, send_request_to, 0,
665  recv_buffer_by_task[get_request_from],
666  n_to_recv[get_request_from] * byte_per_particle,
667  MPI_CHAR, get_request_from, 0,
668  MPI_COMM_WORLD, &status);
669  }
670 
671  // Copy over the particle data (this also updates the total number of particles)
672  copy_over_recieved_data(recv_buffer, ntot_to_recv);
