49  // If no root is found it will throw an error
50  //=======================================================================
51  double find_root_bisection(
52  const std::function<double(double)>& function,
53  std::pair<double, double> xrange,
54  double epsilon)
55  {
56  double x_low = std::min(xrange.first, xrange.second);
57  double x_high = std::max(xrange.first, xrange.second);
58  double y_low = function(x_low);
59  double y_high = function(x_high);
60  const int sign = y_high > y_low ? 1 : -1;
61 
62  // Accuracy we want
63  const double delta_x = (x_high - x_low) * epsilon;
64 
65  // Sanity check
66  if (y_low * y_high > 0.0) {
67  throw std::runtime_error(
68  "Error binary_search_for_value. f(xlow) and f(xhigh) "
69  "have the same sign!");
70  }
71 
72  // Do the binary search
73  int count = 0;
74  while (x_high - x_low > delta_x) {
75  double x_mid = (x_high + x_low) / 2.0;
76  double y_mid = function(x_mid) * sign;
77  if (y_mid > 0.0) {
78  x_high = x_mid;
79  } else {
80  x_low = x_mid;
81  }
82  count++;
83  }
84 
85  return x_low;
86  }

150  {

153  const double chinu2 = chinu * chinu;
154 
155  // Geometry factors (cotc is equivalent of sinc, i.e. cot(x)*x)
156  double cotcKchi = 1.0;
157  double sincKchi = 1.0;
158  if(K == 0.0){
159  cotcKchi = 1.0;
160  sincKchi = 1.0;
161  } else if(K < 0.0) {
162  cotcKchi = chi > 1e-8 ? chi/tanh(chi) : 1.0;
163  sincKchi = chi > 1e-8 ? sinh(chi)/chi : 1.0;
164  } else {
165  cotcKchi = chi > 1e-8 ? chi/tan(chi) : 1.0;
166  sincKchi = chi > 1e-8 ? sin(chi)/chi : 1.0;
167  }
168 
169  // Start the recursion at a large enough lmax such that j_ell/j_ell-1 ~ 0
170  // and bring it down to lmax where we start to store the values
171  int lstart = std::max( lmax , int( lmax < 10 ? 5*chinu : (lmax < 100 ? 1.6*chinu : 1.2*chinu) ) );
172  double h = 0.0;
173  double sqrtnu2chi2old = sqrt(chinu2 - chi*chi*K*(lstart+1)*(lstart+1));
174  for (int k = lstart; k >= lmax+1; k--) {
175  double sqrtnu2chi2 = sqrt(chinu2 - chi*chi*K*k*k);
176  h = sqrtnu2chi2old / ( (2 * k + 1) * cotcKchi - sqrtnu2chi2 * h);
177  sqrtnu2chi2old = sqrtnu2chi2;
178  }
179 
180  // Recursion relation for j_(n+1) / jn
181  DVector res(lmax + 1, 0.0);
182  for (int k = lmax; k >= 1; k--) {
183  double sqrtnu2chi2 = sqrt(chinu2 - chi*chi*K*k*k);
184  h = sqrtnu2chi2old / ( (2 * k + 1) * cotcKchi - sqrtnu2chi2 * h);
185  res[k] = h;
186  sqrtnu2chi2old = sqrtnu2chi2;
187  }

191  for (int ell = 1; ell <= lmax; ell++) {

414  // maxsteps denotes the maximum j before we deem it not to converge
415  std::pair<double,bool> GeneralizedLentzMethod(
416  std::function<double(int)> &a,
417  std::function<double(int)> &b,
418  double epsilon,
419  int maxsteps)
420  {
421 
422  const double tiny = 1e-30;
423

425  if(fabs(b0) < tiny) b0 = tiny;
